25void CommandLine::ParseCommandline(int argc, char* argv[])
26{
27 for (int i = 1; i < argc; ++i)
28 {
29 if (argv[i][0] == '-')
30 {
31 commands_.insert(std::string(&argv[i][1]));
32 }
33 else
34 {
35 std::string arg = argv[i];
36 args_.push_back(arg);
37 }
38 }
39}
40
41bool CommandLine::CommandExists(const char* c)
42{
43 return commands_.find(std::string(c)) != commands_.end();
44}
