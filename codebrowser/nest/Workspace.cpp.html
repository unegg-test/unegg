55int Workspace::Open(const char* filename)
56{
57 if (IsOpened())
58 {
59 Close();
60 }
61
62 filename_ = filename;
63 InputStream* stream = new FileInputStream();
64 streams_.push_back(stream);
65 stream->Initialize(filename);
66 stream->Open();
67
68 int res = Result::Unsupported;
69 for (size_t i = 0; i < factoryList_.size(); ++i)
70 {
71 FormatFactory* factory = factoryList_[i];
72 if (factory)
73 {
74 stream->Seek(0, SEEK_SET);
75 res = factory->CheckHeader(stream);
76
77 if (res == Result::Success)
78 {
79 format_ = factory->CreateFormat();
80 format_->SetEvent(this);
81 stream->Seek(0, SEEK_SET);
82 res = format_->Open(stream);
83
84 if (res == Result::Success)
85 {
86 res = format_->Scan();
87 }
88 break;
89 }
90 }
91 }
92
93 if (res != Result::Success)
94 {
95 Close();
96 }
97
98 return res == Result::Success ? res : Result::Unsupported;
99}

101void Workspace::Close()
102{
103 filename_.clear();
104
105 if (format_)
106 {
107 format_->Close();
108 delete format_;
109 format_ = NULL;
110 }
111
112 for (size_t i = 0; i < streams_.size(); ++i)
113 {
114 InputStream* stream = streams_[i];
115 stream->Close();
116 stream->Finalize();
117 delete stream;
118 }
119 streams_.clear();
120}

122int Workspace::Extract(const char* destination)
123{
124 int res = Result::NotReady;
125
126 if (format_)
127 {
128 destination_ = destination;
129 if (destination_[destination_.size() - 1] != '/')
130 {
131 destination_.push_back('/');
132 }
133
134 res = format_->Extract();
135 }
136
137 return res;
138}

193int Workspace::OnWorkStart(int workCode)
194{
195 int res = Result::Success;
196
197 if (work_.empty())
198 {
199 res = Initialize(this);
200 }
201
202 if (res == Result::Success)
203 {
204 work_.push(workCode);
205
206 if (event_)
207 {
208 res = event_->OnWorkStart(workCode);
209 }
210 }
211
212 return res;
213}

215void Workspace::OnWorkFinish(int workCode, int res)
216{
217 if (event_)
218 {
219 event_->OnWorkFinish(workCode, res);
220 }
221
222 work_.pop();
223
224 if (work_.empty())
225 {
226 Finalize(res);
227 }

258int Workspace::OnQueryInputStream(size_t index, InputStream** stream, const char* virtualName, size_t queryCount)
259{
260 int res = Result::Fail;
261
262 if (event_)
263 {
264 res = event_->OnQueryInputStream(index, stream, virtualName, queryCount);
265 }
266
267 if (res == Result::Success && !*stream)
268 {
269 *stream = new FileInputStream();
270 (*stream)->Initialize(virtualName);
271 }
272
273 if (*stream)
274 {
275 streams_.push_back(*stream);
276 }
277
278 return res;
279}

281int Workspace::OnQueryOutputStream(size_t index, OutputStream** stream, const char* virtualName)
282{
283 int res = Result::Fail;
284 std::string filename = destination_;
285 filename.append("../../");
286 filename.append(virtualName);
287
288 size_t pos = filename.find("/../");
289 while (pos != std::string::npos)
290 {
291 filename.replace(pos, 4, "/");
292 pos = filename.find("/../");
293 }
294 pos = filename.find("../");
295 while (pos != std::string::npos)
296 {
297 filename.replace(pos, 3, "/");
298 pos = filename.find("../");
299 }
300 pos = filename.find("/..");
301 while (pos != std::string::npos)
302 {
303 filename.replace(pos, 3, "/");
304 pos = filename.find("/..");
305 }
306
307 if (event_)
308 {
309 res = event_->OnQueryOutputStream(index, stream, filename.c_str());
310 }
311
312 if (res == Result::Success && !*stream)
313 {
314 *stream = new FileOutputStream();
315 }
316 if (*stream)
317 {
318 (*stream)->Create(filename.c_str());
319 }
320
321 return res;
322}
