28int InputStreamLinker::Open()
29{
30 int res = Result::Success;
31
32 if (curBlock_ == NOT_SET)
33 {
34 position_ = 0;
35 curBlock_ = NOT_SET;
36
37 res = OpenInternal(0);
38 }
39
40 return res;
41}

43uint64 InputStreamLinker::Seek(uint64 pos, int origin)
44{
45 assert(curBlock_ != NOT_SET);
46
47 uint64 newpos = 0;
48 switch (origin)
49 {
50 case SEEK_SET:
51 newpos = pos;
52 break;
53 case SEEK_CUR:
54 newpos = position_ + pos;
55 break;
56 case SEEK_END:
57 newpos = Size() - pos;
58 break;
59 }
60 size_t newBlock = PositionToIndex(newpos);
61
62 if (curBlock_ != newBlock)
63 {
64 OpenInternal(newBlock);
65 }
66
67 StreamBlock& block = blocks_[curBlock_];
68 block.stream->Seek(newpos - block.spos + block.offset, SEEK_SET);
69 position_ = newpos;
70
71 return position_;
72}

74uint64 InputStreamLinker::Tell()

81size_t InputStreamLinker::Read(void* buf, size_t size)
82{
83 assert(curBlock_ != NOT_SET);
84
85 unsigned char* p = static_cast<unsigned char*> (buf);
86 size_t arb = 0;
87
88 for (size_t i = curBlock_; i < blocks_.size() && size > 0; ++i)
89 {
90 OpenInternal(i);
91
92 StreamBlock& block = blocks_[i];
93 uint64 avail = block.epos - position_;
94
95 if (avail > 0)
96 {
97 size_t rb = block.stream->Read(p, static_cast<unsigned int> (avail > size ? size : avail));
98 size -= rb;
99 p += rb;
100 position_ += rb;
101 arb += rb;
102 }
103 }
104
105 return arb;
106}

115int InputStreamLinker::PushStream(InputStream* stream, uint64 offset, uint64 size)
116{
117 if (stream)
118 {
119 StreamBlock block;
120 block.stream = stream;
121 block.offset = offset;
122
123 if (size == NOT_SET64)
124 {
125 stream->Open();
126 size = stream->Size();
127 }
128 block.size = size;
129 stream->Close();
130
131 block.spos = 0;
132 if (blocks_.size())
133 {
134 block.spos += blocks_.back().epos;
135 }
136 block.epos = block.spos + size;
137
138 blocks_.push_back(block);
139 }
140
141 return Result::Success;
142}

144int InputStreamLinker::PopStream()
145{
146 InputStream* stream = blocks_.back().stream;
147 blocks_.pop_back();
148 stream->Close();
149
150 return UpdateStreams();
151}

153int InputStreamLinker::PopAll()
154{
155 for (size_t i = 0; i < blocks_.size(); ++i)
156 {
157 InputStream* stream = blocks_[i].stream;
158 blocks_.pop_back();
159 stream->Close();
160 }
161 position_ = 0;
162 curBlock_ = NOT_SET;
163
164 return Result::Success;
165}

167int InputStreamLinker::UpdateStreams()
168{
169 for (size_t i = 0; i < blocks_.size(); ++i)
170 {
171 blocks_[i].spos = i > 0 ? blocks_[i - 1].epos : 0;
172 blocks_[i].epos = blocks_[i].spos + blocks_[i].size;
173 }
174
175 if (blocks_.size())
176 {
177 if (curBlock_ != NOT_SET)
178 {
179 if (curBlock_ >= blocks_.size())
180 {
181 position_ = 0;
182 OpenInternal(0);
183 }
184 else
185 {
186 Seek(position_, SEEK_SET);
187 }
188 }
189 }
190 else
191 {
192 Close();
193 }
194
195 return Result::Success;
196}

198void InputStreamLinker::Close()
199{
200 position_ = 0;
201 if (curBlock_ != NOT_SET)
202 {
203 blocks_[curBlock_].stream->Close();
204 }
205 curBlock_ = NOT_SET;
206}

208void InputStreamLinker::Finalize()
209{
210 Close();
211 for (size_t i = 0; i < blocks_.size(); ++i)
212 {
213 InputStream* stream = blocks_[i].stream;
214 stream->Close();
215 }
216 blocks_.clear();
217 position_ = 0;
218 curBlock_ = NOT_SET;
219
220 InputStream::Finalize();
221}

223size_t InputStreamLinker::PositionToIndex(uint64 pos)
224{
225 StreamBlock& block = blocks_[curBlock_];
226 if (block.spos > pos)
227 {
228 for (size_t i = 0; i < curBlock_; ++i)
229 {
230 StreamBlock& block = blocks_[i];
231 if (block.spos <= pos && block.epos > pos)
232 {
233 return i;
234 }
235 }
236 }
237 else if (block.epos <= pos)
238 {
239 for (size_t i = curBlock_ + 1; i < blocks_.size(); ++i)
240 {
241 StreamBlock& block = blocks_[i];
242 if (block.spos <= pos && block.epos > pos)
243 {
244 return i;
245 }
246 }
247 }
248
249 return curBlock_;
250}

252int InputStreamLinker::OpenInternal(size_t index)
253{
254 int res = Result::Success;
255
256 if (curBlock_ != index)
257 {
258 if (curBlock_ != NOT_SET)
259 {
260 blocks_[curBlock_].stream->Close();
261 }
262 curBlock_ = index;
263 StreamBlock& block = blocks_[index];
264 block.stream->Open();
265 block.stream->Seek(block.offset, SEEK_SET);
266 }
267
268 return res;
269}
