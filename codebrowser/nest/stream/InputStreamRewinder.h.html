12#ifndef NEST_STREAM_STREAMREWINDER_H_
13#define NEST_STREAM_STREAMREWINDER_H_
14
15#include <stdio.h>
16#include <vector>
17#include <nest/stream/InputStream.h>
18
19namespace nest
20{
21
22class InputStreamRewinder
23{
24public:
25
26 InputStreamRewinder(InputStream& stream) :
27 stream_(&stream)
28 {
29 Push();
30 }
31
32 InputStreamRewinder(InputStream* stream) :
33 stream_(stream)
34 {
35 Push();
36 }
37
38 virtual ~InputStreamRewinder()
39 {
40 if (stack_.size())
41 {
42 stream_->Seek(stack_[0], SEEK_SET);
43 }
44 }
45
46 inline void Push(void)
47 {
48 stack_.push_back(stream_->Tell());
49 }
50
51 inline void Pop(void)
52 {
53 stream_->Seek(stack_.back(), SEEK_SET);
54 stack_.pop_back();
55 }
56
57 inline void Clear(void)
58 {
59 stack_.clear();
60 }
61
62 inline bool IsEmpty()
63 {
64 return stack_.empty();
65 }
66
67private:
68 std::vector<uint64> stack_;
69 InputStream* stream_;
70};
71
72}
73
74#endif /* NEST_STREAM_STREAMREWINDER_H_ */
