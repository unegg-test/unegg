34int FileOutputStream::Create(const char* virtualName)
35{
36 Close();
37
38 int res = EnsureFilePath(virtualName);
39
40 if (res == Result::Success)
41 {
42 file_ = fopen(virtualName, "wb");
43 res = file_ != NULL ? Result::Success : Result::WriteError;
44
45 if (res == Result::Success)
46 {
47 filename_ = virtualName;
48 }
49 }
50
51 return res;
52}

73int FileOutputStream::EnsureFilePath(const char* path)
74{
75 int res = Result::Success;
76 char* tmp = const_cast<char*> (path);
77
78 char* pos = strrchr(tmp, '/');
79 if (pos)
80 {
81 char backup = *pos;
82 *pos = 0;
83 res = ForceDirectory(path);
84 *pos = backup;
85 }
86 else
87 {
88 res = ForceDirectory(path);
89 }
90
91 return res;
92}
93
94int FileOutputStream::ForceDirectory(const char* path)
95{
96 int res = Result::Success;
97 bool exists = 0 == access(path, F_OK);
98 if (!exists)
99 {
100 res = mkdir_r(path, 0755) != 0 ? errno : Result::Success;
101 if (errno == EEXIST)
102 {
103 res = Result::Success;
104 }
105 }
106
107 if (res != Result::Success)
108 {
109 printf("\ncode : %d\n", errno);
110 }
111
112 return res;
113}
