29int FileInputStream::Open()
30{
31 Close();
32 size_ = -1;
33
34 file_ = fopen(virtualName.c_str(), "rb");
35 if (file_ != NULL)
36 {
37 Size();
38 }
39
40 return file_ != NULL ? Result::Success : Result::ReadError;
41}
42
43void FileInputStream::Close()
44{
45 if (file_)
46 {
47 fclose(file_);
48 file_ = NULL;
49 }
50}

52uint64 FileInputStream::Seek(uint64 pos, int origin)
53{
54 return fseeko64(file_, pos, origin);
55}
56
57uint64 FileInputStream::Tell()
58{
59 return ftello(file_);
60}
61
62size_t FileInputStream::Read(void* buf, size_t size)
63{
64 return fread(buf, 1, size, file_);
65}
66
67uint64 FileInputStream::Size()
68{
69 if (size_ == static_cast<uint64> (-1))
70 {
71 uint64 offset = Tell();
72 Seek(0, SEEK_END);
73 size_ = Tell();
74 Seek(offset, SEEK_SET);
75 }
76
77 return size_;
78}
