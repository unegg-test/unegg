23class InputStreamLinker: public InputStream
24{
25public:
26 InputStreamLinker();
27 virtual ~InputStreamLinker();
28
29 virtual int Open();
30 virtual void Close();
31 virtual void Finalize();
32
33 virtual uint64 Seek(uint64 pos, int origin);

46 uint64 size;

48 uint64 epos;
49 };
50
51 StreamBlock* GetBlock(size_t index);

54
55 static const size_t NOT_SET = (size_t) (-1);
56 static const uint64 NOT_SET64 = (uint64) (-1);
57
58protected:
59 size_t PositionToIndex(uint64 pos);
60 int OpenInternal(size_t index);
61
62private:
63 uint64 position_;
64 size_t curBlock_;
65 std::vector<StreamBlock> blocks_;
66};
67
68}
