39int ExtraField::OnRead(InputStreamRewinder* rewinder, InputStream* stream)
40{
41 int res = Result::ReadError;
42
43 IF_READ_STREAM_VAR(stream, signature_)
44 {
45 if (signature_ == Magic::END)
46 {
47 res = Result::End;
48 rewinder->Clear();
49 }
50 else
51 {
52
53 IF_READ_STREAM_VAR(stream, gpb_)
54 {
55 unsigned int ls = 0;
56 unsigned short ss = 0;
57
58 if ((gpb_ & 1) == 1)
59 {
60
61 IF_READ_STREAM_VAR(stream, ls)
62 {
63 size_ = ls;
64 }
65 else
66 {
67 return res;
68 }
69 }
70 else
71 {
72
73 IF_READ_STREAM_VAR(stream, ss)
74 {
75 size_ = ss;
76 }
77 else
78 {
79 return res;
80 }
81 }
82
83 if (size_)
84 {
85 // for filename and comment..
86 data_ = new char[size_ + 1];
87 data_[size_] = 0;
88
89 IF_READ_STREAM_PTR(stream, data_, size_)
90 {
91 res = Result::Success;
92 rewinder->Clear();
93 }
94 else
95 {
96 delete[] data_;
97 }
98 }
99 else
100 {
101 res = Result::Success;
102 rewinder->Clear();
103 }
104 }
105 }
106 }
107
108 return res;
109}
