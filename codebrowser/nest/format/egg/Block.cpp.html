33int Block::OnRead(InputStreamRewinder* rewinder, InputStream* stream)
34{
35 int res = Result::ReadError;
36 int sig = 0;
37
38 IF_READ_STREAM_VAR(stream, sig)
39 {
40 if (sig == Magic::BLOCK)
41 {
42
43 IF_READ_STREAM_VAR(stream, compMethod_)
44 {
45
46 IF_READ_STREAM_VAR(stream, methodHint_)
47 {
48
49 IF_READ_STREAM_VAR(stream, uncompSize_)
50 {
51
52 IF_READ_STREAM_VAR(stream, compSize_)
53 {
54
55 IF_READ_STREAM_VAR(stream, crc_)
56 {
57 rewinder->Clear();
58 dataOffset_ = stream->Tell();
59
60 res = Result::Success;
61 }
62 }
63 }
64 }
65 }
66 }
67 }
68
69 return res;
70}

72int Block::_As4Bytes(unsigned int key, bool* handled)
73{
74 VERIFY_KEY(key::block::size::PACKED, GetCompressedSize());
75 VERIFY_KEY(key::block::size::UNPACKED, GetUncompressedSize());
76 VERIFY_KEY(key::block::method::HINT, GetMethodHint());
77 VERIFY_KEY(key::block::method::METHOD, GetCompressMethod());
78 VERIFY_KEY(key::block::CRC, GetCRC());
79
80 return Header::_As4Bytes(key, handled);
81}

83int64 Block::_As8Bytes(unsigned int key, bool* handled)

90const char* Block::GetComressMethodString()
91{
92 switch (compMethod_)
93 {
94 case CompressionMethod::Store:
95 return "STORE";
96 case CompressionMethod::Deflate:
97 return "DEFLATE";
98 case CompressionMethod::bzip:
99 return "BZIP";
100 case CompressionMethod::lzma:
101 return "LZMA";
102 case CompressionMethod::azo:
103 return "AZO";
104 }
105
106 return "";
107}

109const void* Block::_AsPtr(unsigned int key, bool* handled)

111 VERIFY_KEY(key::block::method::NAME, GetComressMethodString());
