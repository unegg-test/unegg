12#ifndef NEST_FORMAT_EGG_COMMENTFIELD_H_
13#define NEST_FORMAT_EGG_COMMENTFIELD_H_
14
15#include <nest/format/egg/ExtraField.h>
16#include <string>
17
18namespace nest
19{
20namespace egg
21{
22
23class CommentField: public ExtraField
24{
25public:
26 inline bool IsEncrypted()
27 {
28 return GetBitFlag(3);
29 }
30
31 inline bool IsUTF8()
32 {
33 return GetBitFlag(4);
34 }
35
36 inline unsigned short GetLocale()
37 {
38 return IsUTF8() ? -1 : *(unsigned short*) GetDataPtr();
39 }
40
41 inline const char* GetComment()
42 {
43 return GetDataPtr(2 - (IsUTF8() ? 2 : 0));
44 }
45};
46
47}
48}
49
50#endif /* NEST_FORMAT_EGG_COMMENTFIELD_H_ */
