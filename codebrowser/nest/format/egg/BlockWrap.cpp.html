31void BlockWrap::OnCleanUp()

34 extras_.Clear();
35 dataOffset_ = 0;
36}

38int BlockWrap::OnRead(InputStreamRewinder* rewinder, InputStream* stream)
39{
40 int res = block_.Read(stream);
41
42 if (res == Result::Success)
43 {
44 res = extras_.Read(stream);
45
46 if (res == Result::Success)
47 {
48 dataOffset_ = stream->Tell();
49 rewinder->Clear();
50 }
51 }
52 else
53 {
54 res = Result::End;
55 }
56
57 return res;
58}

60int BlockWrap::_As4Bytes(unsigned int key, bool* handled)
61{
62 bool handle = false;
63 int res = block_.AsInt32(key, &handle);
64 if (!handle)
65 {
66 res = extras_.AsInt32(key, &handle);
67 }
68 if (handled)
69 {
70 *handled = handle;
71 }
72 assert(handled || handle);
73
74 return res;
75}
