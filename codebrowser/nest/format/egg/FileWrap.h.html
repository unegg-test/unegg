15#include <vector>
16#include <nest/Info.h>
17#include <nest/format/Header.h>
18#include <nest/format/egg/ExtraFieldWrap.h>
19#include <nest/format/egg/File.h>
20#include <nest/format/egg/BlockWrap.h>

24namespace nest
25{
26namespace egg
27{
28
29class FileWrap : public Header
30{
31public:
32 FileWrap(bool isSolid);
33 virtual ~FileWrap();
34
35 inline File& GetFileHeader()
36 {
37 return file_;
38 }
39
40 inline ExtraFieldWrap& GetExtras()
41 {
42 return extras_;
43 }
44
45 inline ExtraField* GetExtra(int magic)
46 {
47 return extras_.GetExtra(magic);
48 }
49
50 inline size_t GetBlockCount()
51 {
52 return blocks_.size();
53 }
54
55 inline BlockWrap* GetBlock(size_t index)
56 {
57 return blocks_[index];
58 }
59
60protected:
61 virtual void OnCleanUp();
62 virtual int OnRead(InputStreamRewinder* rewinder, InputStream* stream);
63
64 // Info
65 virtual int _As4Bytes(unsigned int key, bool* handled);
66 virtual int64 _As8Bytes(unsigned int key, bool* handled);
67 virtual size_t _AsSize(unsigned int key, bool* handled);
68 virtual const void* _AsPtr(unsigned int key, bool* handled);
69 virtual Info* _AsInfo(unsigned int key, bool* handled);
70 virtual Info* _AsChildInfo(unsigned int key, size_t index, bool* handled);
71
72private:
73 File file_;
74 ExtraFieldWrap extras_;
75 std::vector<BlockWrap*> blocks_;
76 bool isSolid_;
77};
78
79}
80}
