33void FileWrap::OnCleanUp()
34{
35 file_.Clear();
36 extras_.Clear();
37
38 if (blocks_.size())
39 {
40 for (size_t i = 0; i < blocks_.size(); ++i)
41 {
42 delete blocks_[i];
43 }
44 blocks_.clear();
45 }
46}

48int FileWrap::OnRead(InputStreamRewinder* rewinder, InputStream* stream)
49{
50 int res = file_.Read(stream);
51
52 if (res == Result::Success)
53 {
54 res = extras_.Read(stream);
55
56 if (res == Result::Success)
57 {
58 if (isSolid_)
59 {
60 rewinder->Clear();
61 }
62 else
63 {
64 for (size_t i = 0; i < MAX_BLOCKS; ++i)
65 {
66 BlockWrap* bh = new BlockWrap();
67 res = bh->Read(stream);
68
69 switch (res)
70 {
71 case Result::End:
72 rewinder->Clear();
73 delete bh;
74 return Result::Success;
75 case Result::Success:
76 blocks_.push_back(bh);
77 stream->Seek(bh->GetBlock().GetCompressedSize(), SEEK_CUR);
78 break;
79 default:
80 delete bh;
81 return res;
82 }
83 }
84 }
85 }
86 }
87 else
88 {
89 res = Result::End;
90 }
91
92 return res;
93}

95int FileWrap::_As4Bytes(unsigned int key, bool* handled)
96{
97 if (blocks_.size())
98 {
99 VERIFY_KEY(key::file::CRC, blocks_[0]->AsInt32(key::block::CRC, handled));
100 }
101
102 bool handle = false;
103 int res = file_.AsInt32(key, &handle);
104 if (!handle)
105 {
106 res = extras_.AsInt32(key, &handle);
107 }
108 if (handled)
109 {
110 *handled = handle;
111 }
112 assert(handled || handle);
113
114 return res;
115}

117int64 FileWrap::_As8Bytes(unsigned int key, bool* handled)
118{

121 uint64 size = 0;

123 {
124 size += blocks_[i]->AsUInt32(key::block::size::PACKED);
