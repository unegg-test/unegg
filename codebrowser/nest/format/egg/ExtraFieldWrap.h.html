12#ifndef NEST_FORMAT_EGG_EXTRAFIELDWRAP_H_
13#define NEST_FORMAT_EGG_EXTRAFIELDWRAP_H_
14
15#include <nest/format/Header.h>
16#include <nest/format/egg/ExtraField.h>
17
18namespace nest
19{
20namespace egg
21{
22
23class ExtraFieldWrap: public Header
24{
25public:
26 ExtraFieldWrap();
27 virtual ~ExtraFieldWrap();
28
29 inline bool FieldExists(int magic)
30 {
31 return GetExtra(magic) != NULL;
32 }
33
34 inline ExtraField* GetExtra(int magic)
35 {
36 return extras_[MagicToIndex(magic)];
37 }
38
39protected:
40 inline int MagicToIndex(int magic)
41 {
42 return magic / 67 % 7;
43 }
44
45 virtual void OnCleanUp();
46 virtual int OnRead(InputStreamRewinder* rewinder, InputStream* stream);
47
48 // Info
49 virtual int _As4Bytes(unsigned int key, bool* handled);
50 virtual int64 _As8Bytes(unsigned int key, bool* handled);
51 virtual size_t _AsSize(unsigned int key, bool* handled);
52 virtual const void* _AsPtr(unsigned int key, bool* handled);
53
54private:
55 ExtraField* extras_[7];
56};
57
58}
59}
60
61#endif /* NEST_FORMAT_EGG_EXTRAFIELDWRAP_H_ */
