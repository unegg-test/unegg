12#ifndef NEST_FORMAT_EGG_FILENAMEFIELD_H_
13#define NEST_FORMAT_EGG_FILENAMEFIELD_H_
14
15#include <nest/format/egg/ExtraField.h>
16
17namespace nest
18{
19namespace egg
20{
21
22class FilenameField: public ExtraField
23{
24public:
25 inline bool IsEncrypted()
26 {
27 return GetBitFlag(3);
28 }
29
30 inline bool IsUTF8()
31 {
32 return !GetBitFlag(4);
33 }
34
35 inline bool IsAbsolute()
36 {
37 return GetBitFlag(5);
38 }
39
40 inline unsigned short GetLocale()
41 {
42 return IsUTF8() ? -1 : *(unsigned short*) GetDataPtr();
43 }
44
45 inline unsigned int GetParentPathID()
46 {
47 return IsAbsolute() ? *(unsigned int*) GetDataPtr(2 - (IsUTF8() ? 2 : 0)) : -1;
48 }
49
50 inline const char* GetFileName()
51 {
52 return GetDataPtr((IsAbsolute() ? 6 : 2) - (IsUTF8() ? 2 : 0));
53 }
54};
55
56}
57}
58
59#endif /* NEST_FORMAT_EGG_FILENAMEFIELD_H_ */
