32int File::OnRead(InputStreamRewinder* rewinder, InputStream* stream)
33{
34 int res = Result::ReadError;
35 int sig = 0;
36
37 IF_READ_STREAM_VAR(stream, sig)
38 {
39 if (sig == Magic::FILE)
40 {
41
42 IF_READ_STREAM_VAR(stream, id_)
43 {
44
45 IF_READ_STREAM_VAR(stream, length_)
46 {
47 rewinder->Clear();
48 res = Result::Success;
49 }
50 }
51 }
52 }
53
54 return res;
55}
