42 switch (GetEncryptMethod())
43 {
44 default:
45 return NULL;
46 case 1:
47 case 2:

50 }
51
52 inline const char* GetAESFooter()

54 switch (GetEncryptMethod())
55 {
56 default:
57 return NULL;
