37int Global::OnRead(InputStreamRewinder* rewinder, InputStream* stream)
38{
39 int res = Result::ReadError;
40 int sig = 0;
41
42 IF_READ_STREAM_VAR(stream, sig)
43 {
44 if (sig == Magic::EGG)
45 {
46
47 IF_READ_STREAM_VAR(stream, version_)
48 {
49
50 IF_READ_STREAM_VAR(stream, id_)
51 {
52
53 IF_READ_STREAM_VAR(stream, reserved_)
54 {
55 rewinder->Clear();
56 res = Result::Success;
57 }
58 }
59 }
60 }
61 }
62
63 return res;
64}

66int Global::_As4Bytes(unsigned int key, bool* handled)
67{
68 VERIFY_KEY(key::archive::version::MAJOR, GetVersion() >> 8);
69 VERIFY_KEY(key::archive::version::MINOR, GetVersion() & 0xFF);
70 VERIFY_KEY(key::archive::volume::ID, GetID());
71
72 return Header::_As4Bytes(key, handled);
73}
