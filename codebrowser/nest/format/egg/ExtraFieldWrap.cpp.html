49int ExtraFieldWrap::OnRead(InputStreamRewinder* rewinder, InputStream* stream)
50{
51 assert(rewinder && stream);
52
53 int res = Result::Invalid;
54
55 for (size_t i = 0; i < MAX_HEADERS; ++i)
56 {
57 ExtraField* eh = new ExtraField();
58 res = eh->Read(stream);
59 if (res == Result::Success && eh->GetSignature() != Magic::END)
60 {
61 switch (eh->GetSignature())
62 {
63 case Magic::SPLIT:
64 case Magic::SOLID:
65 case Magic::ENCRYPT:
66 case Magic::FILENAME:
67 case Magic::COMMENT:
68 case Magic::WINDOWS_FILEINFO:
69 case Magic::POSIX_FILEINFO:
70 extras_[MagicToIndex(eh->GetSignature())] = eh;
71 break;
72 default:
73 delete eh;
74 }
75 }
76 else
77 {
78 if (eh->GetSignature() == Magic::END)
79 {
80 rewinder->Clear();
81 res = Result::Success;
82 }
83
84 delete eh;
85 break;
86 }
87 }
88
89 return res;
90}

92int ExtraFieldWrap::_As4Bytes(unsigned int key, bool* handled)
93{
94 CommentField* cfield = (CommentField*)extras_[MagicToIndex(Magic::COMMENT)];
95 EncryptField* efield = (EncryptField*)extras_[MagicToIndex(Magic::ENCRYPT)];
96 FilenameField* ffield = (FilenameField*)extras_[MagicToIndex(Magic::FILENAME)];
97 PosixField* pfield = (PosixField*)extras_[MagicToIndex(Magic::POSIX_FILEINFO)];
98 WindowsField* wfield = (WindowsField*)extras_[MagicToIndex(Magic::WINDOWS_FILEINFO)];
99 SplitField* sfield = (SplitField*)extras_[MagicToIndex(Magic::SPLIT)];
100
101 VERIFY_KEY(key::file::IS_ENCRYPTED, efield ? 1 : 0);
102
103 if (cfield)
104 {
105 VERIFY_KEY(key::file::comment::LOCALE, cfield->GetLocale());
106 VERIFY_KEY(key::file::comment::IS_UTF8, cfield->IsUTF8());
107 VERIFY_KEY(key::file::comment::IS_ENCRYPTED, cfield->IsEncrypted());
108 VERIFY_KEY(key::archive::comment::LOCALE, cfield->GetLocale());
109 VERIFY_KEY(key::archive::comment::IS_UTF8, cfield->IsUTF8());
110 VERIFY_KEY(key::archive::comment::IS_ENCRYPTED, cfield->IsEncrypted());
111 }
112
113 if (efield)
114 {
115 VERIFY_KEY(key::file::encrypt::METHOD, efield->GetEncryptMethod());
116 VERIFY_KEY(key::file::encrypt::CRC, efield->GetCRC());
117 VERIFY_KEY(key::archive::encrypt::METHOD, efield->GetEncryptMethod());
118 VERIFY_KEY(key::archive::encrypt::CRC, efield->GetCRC());
119 VERIFY_KEY(key::block::encrypt::METHOD, efield->GetEncryptMethod());
120 VERIFY_KEY(key::block::encrypt::CRC, efield->GetCRC());
121 }
122
123 if (ffield)
124 {
125 VERIFY_KEY(key::file::name::IS_ABSOLUTE, ffield->IsAbsolute());
126 VERIFY_KEY(key::file::name::IS_ENCRYPTED, ffield->IsEncrypted());
127 VERIFY_KEY(key::file::name::IS_UTF8, ffield->IsUTF8());
128 VERIFY_KEY(key::file::name::LOCALE, ffield->GetLocale());
129 VERIFY_KEY(key::file::name::PARENT_ID, ffield->GetParentPathID());
130 }
131
132 if (pfield)
133 {
134 VERIFY_KEY(key::file::attribute::POSIX_MODE, pfield->GetMode());
135 VERIFY_KEY(key::file::attribute::UID, pfield->GetUID());
136 VERIFY_KEY(key::file::attribute::GID, pfield->GetGID());
137 }
138
139 if (wfield)
140 {
141 VERIFY_KEY(key::file::attribute::ATTRIBUTES, wfield->GetAttribute());
142 }
143
144 if (sfield)
145 {
146 VERIFY_KEY(key::archive::volume::NEXT, sfield->GetNextFileID());
147 VERIFY_KEY(key::archive::volume::PREV, sfield->GetPrevFileID());
148 }
149
150 return Header::_As4Bytes(key, handled);
151}

193const void* ExtraFieldWrap::_AsPtr(unsigned int key, bool* handled)
194{
195 CommentField* cfield = (CommentField*)extras_[MagicToIndex(Magic::COMMENT)];
196 EncryptField* efield = (EncryptField*)extras_[MagicToIndex(Magic::ENCRYPT)];
197 FilenameField* ffield = (FilenameField*)extras_[MagicToIndex(Magic::FILENAME)];
198 //PosixField* pfield = (PosixField*)extras_[MagicToIndex(Magic::POSIX_FILEINFO)];
199 //WindowsField* wfield = (WindowsField*)extras_[MagicToIndex(Magic::WINDOWS_FILEINFO)];
200 //SplitField* sfield = (SplitField*)extras_[MagicToIndex(Magic::SPLIT)];
201
202 if (cfield)
203 {
204 VERIFY_KEY(key::archive::comment::COMMENT, cfield->GetComment());
205 VERIFY_KEY(key::file::comment::COMMENT, cfield->GetComment());
206 }
207
208 if (efield)
209 {
210 VERIFY_KEY(key::archive::encrypt::VERIFY_DATA, efield->GetVerifyData());
211 VERIFY_KEY(key::archive::encrypt::HEADER, efield->GetAESHeader());
212 VERIFY_KEY(key::archive::encrypt::FOOTER, efield->GetAESFooter());
213 VERIFY_KEY(key::file::encrypt::VERIFY_DATA, efield->GetVerifyData());
214 VERIFY_KEY(key::file::encrypt::HEADER, efield->GetAESHeader());
215 VERIFY_KEY(key::file::encrypt::FOOTER, efield->GetAESFooter());
216 }
217
218 if (ffield)
219 {
220 VERIFY_KEY(key::file::name::NAME, ffield->GetFileName());
221 }
222
223 return Header::_AsPtr(key, handled);
224}
