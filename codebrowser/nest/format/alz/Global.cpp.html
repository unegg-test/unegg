65int Global::_As4Bytes(unsigned int key, bool* handled)
66{
67 VERIFY_KEY(key::archive::version::MAJOR, version_ >> 8);
68 VERIFY_KEY(key::archive::version::MINOR, version_ & 0xFF);
69 VERIFY_KEY(key::archive::volume::ID, id_);
70
71 return Header::_As4Bytes(key, handled);
72}
