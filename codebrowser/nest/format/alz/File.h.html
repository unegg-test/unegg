12#ifndef NEST_FORMAT_ALZ_FILE_H_
13#define NEST_FORMAT_ALZ_FILE_H_
14
15#include <nest/format/Header.h>
16
17namespace nest
18{
19namespace alz
20{
21
22class File: public Header
23{
24public:
25 File();
26 virtual ~File();
27
28 inline unsigned int GetSignature()
29 {
30 return *(unsigned int*) buffer_;
31 }
32
33 inline unsigned short GetFilenameLength()
34 {
35 return *(unsigned short*) &buffer_[4];
36 }
37
38 inline unsigned char GetAttribute()
39 {
40 return *(unsigned char*) &buffer_[6];
41 }
42
43 inline unsigned int GetDateTime()
44 {
45 return *(unsigned int*) &buffer_[7];
46 }
47
48 inline unsigned short GetBitFlags()
49 {
50 return *((unsigned short*) &buffer_[11]);
51 }
52
53 inline bool IsEncrypted()
54 {
55 return (GetBitFlags() & 1) == 1;
56 }
57
58 inline bool IsCommentExists()
59 {
60 return (GetBitFlags() & 8) == 1;
61 }
62
63 inline unsigned char GetBaseSize()
64 {
65 return (GetBitFlags() & 0xF0) >> 4;
66 }
67
68 inline unsigned short GetCompressMethod()
69 {
70 return compressMethod_ == 2 ? CompressionMethod::Deflate : CompressionMethod::Store;
71 }
72
73 inline unsigned int GetCRC()
74 {
75 return crc_;
76 }
77
78 inline uint64 GetPackedSize()
79 {
80 return packedSize_;
81 }
82
83 inline uint64 GetUnpackedSize()
84 {
85 return unpackedSize_;
86 }
87
88 inline uint64 GetDataOffset()
89 {
90 return offset_;
91 }
92
93 inline const char* GetFilename()
94 {
95 return filename_.c_str();
96 }
97
98 inline const char* GetComment()
99 {
100 return comment_.c_str();
101 }
102
103 inline void SetComment(const char* comment)
104 {
105 comment_ = comment;
106 }
107
108 inline const char* GetVerifyData()
109 {
110 return verifyData_;
111 }
112
113protected:
114 int ReadFileSize(InputStream* stream, uint64* size, size_t byteCount);
115
116 virtual void OnCleanUp();
117 virtual int OnRead(InputStreamRewinder* rewinder, InputStream* stream);
118
119 virtual int _As4Bytes(unsigned int key, bool* handled);
120 virtual int64 _As8Bytes(unsigned int key, bool* handled);
121 virtual size_t _AsSize(unsigned int key, bool* handled);
122 virtual const void* _AsPtr(unsigned int key, bool* handled);
123
124 const char* GetCompressMethodString();
125
126private:
127 unsigned char buffer_[13];
128 unsigned int crc_;
129 unsigned short compressMethod_;
130 uint64 packedSize_;
131 uint64 unpackedSize_;
132 uint64 offset_;
133 char* verifyData_;
134 std::string filename_;
135 std::string comment_;
136};
137
138}
139}
140
141#endif /* NEST_FORMAT_ALZ_FILE_H_ */
