50int File::ReadFileSize(InputStream* stream, uint64* size, size_t byteCount)
51{
52 int res = Result::ReadError;
53
54 char buffer[8];
55 size_t read = stream->Read(buffer, byteCount);
56 if (read == byteCount)
57 {
58 res = Result::Success;
59 switch (byteCount)
60 {
61 case 1:
62 *size = *(unsigned char*) (buffer);
63 break;
64 case 2:
65 *size = *(unsigned short*) (buffer);
66 break;
67 case 4:
68 *size = *(unsigned int*) (buffer);
69 break;
70 case 8:
71 *size = *(uint64*) (buffer);
72 break;
73 default:
74 res = Result::Invalid;
75 }
76 }
77
78 return res;
79}

81int File::OnRead(InputStreamRewinder* rewinder, InputStream* stream)
82{
83 int res = Result::ReadError;
84
85 size_t read = stream->Read(buffer_, 4);
86 if (read == 4)
87 {
88 switch (GetSignature())
89 {
90 case Magic::FILE:
91 {
92 read = stream->Read(buffer_ + 4, 9);
93 if (read == 9)
94 {
95 packedSize_ = 0;
96 unpackedSize_ = 0;
97
98 if (GetBitFlags() != 0)
99 {
100 read = stream->Read(&compressMethod_, 2);
101 if (read == 2)
102 {
103 read = stream->Read(&crc_, 4);
104 if (read == 4)
105 {
106 size_t len = GetBaseSize();
107 res = ReadFileSize(stream, &packedSize_, len);
108 if (res == Result::Success)
109 {
110 res = ReadFileSize(stream, &unpackedSize_, len);
111 }
112 }
113 }
114 }
115 else
116 {
117 res = Result::Success;
118 }
119
120 if (res == Result::Success)
121 {
122 size_t len = GetFilenameLength();
123 char* fnbuf = new char[len + 1];
124 read = stream->Read(fnbuf, len);
125 fnbuf[len] = 0;
126
127 if (read == len)
128 {
129 utf8::euckr2utf8(fnbuf, &filename_);
130 offset_ = stream->Tell();
131 if (stream->Tell() + GetPackedSize() > stream->Size())
132 {
133 res = Result::NeedMoreStream;
134 }
135 else
136 {
137 if (IsEncrypted())
138 {
139 verifyData_ = new char[12];
140 stream->Read(verifyData_, 12);
141 offset_ += 12;
142 }
143 stream->Seek(GetPackedSize(), SEEK_CUR);
144 res = Result::Success;
145 }
146 rewinder->Clear();
147 }
148 }
149
150 if (!rewinder->IsEmpty())
151 {
152 filename_.clear();
153 }
154 }
155 }
156 break;
157 case Magic::COMMENT:
158 case Magic::END:
159 case Magic::SPLIT:
160 case Magic::FILE_END:
161 res = Result::End;
162 break;
163 }
164 }
165
166 return res;
167}

169int File::_As4Bytes(unsigned int key, bool* handled)
170{
171 VERIFY_KEY(key::file::attribute::ATTRIBUTES, GetAttribute());
172 VERIFY_KEY(key::file::attribute::MODIFIED, GetDateTime());
173 VERIFY_KEY(key::file::IS_ENCRYPTED, IsEncrypted());
174 VERIFY_KEY(key::file::method::METHOD, GetCompressMethod());
175 VERIFY_KEY(key::file::CRC, GetCRC());
176
177 return Info::AsInt32(key, handled);
178}

180int64 File::_As8Bytes(unsigned int key, bool* handled)

182 VERIFY_KEY(key::file::size::PACKED, GetPackedSize());
183 VERIFY_KEY(key::file::size::UNPACKED, GetUnpackedSize());
184 VERIFY_KEY(key::file::OFFSET, GetDataOffset());
185
186 return Info::AsInt64(key, handled);
187}
