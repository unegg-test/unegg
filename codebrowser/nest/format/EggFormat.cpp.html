50int EggFormat::Open(InputStream* stream)
51{
52 if (opened_)
53 {
54 Close();
55 }
56
57 istream_.Initialize(stream->GetVirtualName());
58 istream_.PushStream(stream);
59 int res = istream_.Open();
60
61 if (res == Result::Success)
62 {
63 res = CheckHeaderInternal(&istream_, &globalHeader_);
64
65 if (res == Result::Success)
66 {
67 if (globalHeader_.GetVersion() > 0x0100)
68 {
69 res = OnError(SIZE_T_MAX, Result::Unsupported);
70 }
71 }
72 }
73
74 opened_ = res == Result::Success;
75
76 return res;
77}

90int EggFormat::Scan()
91{
92 int res = Result::NotReady;
93
94 if (opened_)
95 {
96 res = OnWorkStart(WorkCode::Scan);
97
98 if (res == Result::Success)
99 {
100 res = ScanHeaders();
101
102 if (res == Result::Success)
103 {
104 isSpanned_ = headerExtras_.GetExtra(egg::Magic::SPLIT) != NULL;
105 isSolid_ = headerExtras_.GetExtra(egg::Magic::SOLID) != NULL;
106
107 if (isSpanned_)
108 {
109 PrepareSpannedStreams();
110 }
111
112 res = ScanFiles();
113 if (res == Result::Success)
114 {
115 res = ScanFooters();
116 }
117 }
118 }
119 OnWorkFinish(WorkCode::Scan, res);
120 }
121
122 return res;
123}

145{

160

166int EggFormat::ScanHeaders(InputStream* stream, egg::ExtraFieldWrap* header)
167{
168 if (!stream)
169 {
170 stream = &istream_;
171 }
172 if (!header)
173 {
174 header = &headerExtras_;
175 }
176
177 stream->Seek(14, SEEK_SET);
178 InputStreamRewinder rewinder(stream);
179
180 int res = header->Read(stream);
181
182 if (res == Result::Success)
183 {
184 rewinder.Clear();
185 }
186
187 return res;
188}

190int EggFormat::ScanFiles()
191{
192 InputStreamRewinder rewinder(istream_);
193 int res = Result::Invalid;
194
195 for (;;)
196 {
197 egg::FileWrap* file = new egg::FileWrap(isSolid_);
198 res = file->Read(&istream_);
199
200 if (res == Result::Success)
201 {
202 files_.push_back(file);
203 res = OnFileStart(files_.size() - 1);
204 OnFileFinish(files_.size() - 1, Result::Success);
205
206 if (res != Result::Success)
207 {
208 break;
209 }
210 }
211 else if (res == Result::End)
212 {
213 delete file;
214 res = Result::Success;
215
216 break;
217 }
218 else
219 {
220 delete file;
221 break;
222 }
223 }
224
225 if (res == Result::Success)
226 {
227 if (isSolid_)
228 {
229 res = ReadSolidBlocks();
230 }
231
232 if (res == Result::Success)
233 {
234 rewinder.Clear();
235 }
236 }
237
238 return res;
239}

268int EggFormat::PrepareSpannedStreams(void)
269{
270 std::vector<SpannedVolume> volumes;
271 InputStreamRewinder sr(istream_);
272
273 int res = Result::Invalid;
274 egg::SplitField* sh = (egg::SplitField*) headerExtras_.GetExtra(egg::Magic::SPLIT);
275
276 if (sh)
277 {
278 size_t qc = 0;
279 size_t ai = 0;
280
281 unsigned int nextID = 0;
282
283 res = Result::Success;
284 while (res == Result::Success)
285 {
286 ++qc;
287 InputStream* stream = NULL;
288 char volumename[FILENAME_LENGTH_MAX] = { 0, };
289 GenerateVolumeName(ai, volumename, FILENAME_LENGTH_MAX);
290 res = OnQueryInputStream(ai, &stream, volumename, qc);
291
292 if (res != Result::Success)
293 {
294 break;
295 }
296
297 if (stream)
298 {
299 stream->Open();
300 egg::Global gh;
301 res = CheckHeaderInternal(stream, &gh);
302
303 if (res == Result::Success)
304 {
305 stream->Seek(0, SEEK_SET);
306 egg::ExtraFieldWrap extra;
307 res = ScanHeaders(stream, &extra);
308
309 if (res == Result::Success)
310 {
311 res = Result::Invalid;
312 egg::SplitField* field = (egg::SplitField*) extra.GetExtra(egg::Magic::SPLIT);
313 if (field && ((ai == 0 && field->GetPrevFileID() == 0) || (ai && gh.GetID() == nextID)))
314 {
315 SpannedVolume vol;
316 vol.stream = stream;
317 vol.offset = ai ? stream->Tell() : 0;
318 vol.size = stream->Size() - vol.offset - CalculateDummySize(stream);
319 volumes.push_back(vol);
320
321 res = Result::Success;
322
323 nextID = field->GetNextFileID();
324 ++ai;
325 qc = 0;
326
327
328 if (!nextID)
329 {
330 break;
331 }
332 }
333 }
334 }
335 stream->Close();
336 }
337 }
338 }
339
340 if (res == Result::Success && volumes.size())
341 {
342 istream_.Close();
343 istream_.PopAll();
344
345 for (size_t i = 0; i < volumes.size(); ++i)
346 {
347 SpannedVolume& vol = volumes[i];
348 istream_.PushStream(vol.stream, vol.offset, vol.size);
349 }
350
351 istream_.UpdateStreams();
352 res = istream_.Open();
353 }
354
355 return res;
356}

358int EggFormat::ReadSolidBlocks()
359{
360 int res = Result::Success;
361
362 solidBlocks_.clear();
363
364 for (size_t i = 0; i < egg::MAX_BLOCKS; ++i)
365 {
366 egg::BlockWrap* bh = new egg::BlockWrap();
367 res = bh->Read(&istream_);
368
369 switch (res)
370 {
371 case Result::End:
372 delete bh;
373 return Result::Success;
374 case Result::Success:
375 solidBlocks_.push_back(bh);
376 istream_.Seek(bh->GetBlock().GetCompressedSize(), SEEK_CUR);
377 break;
378 default:
379 delete bh;
380 return res;
381 }
382 }
383
384 return res;
385}

387unsigned int EggFormat::CalculateDummySize(InputStream* stream)
388{
389 // scan dummy header
390 const int DUMMY_SEARCH_SIZE = 1024;
391
392 stream->Seek(DUMMY_SEARCH_SIZE, SEEK_END);
393 char buffer[DUMMY_SEARCH_SIZE];
394 stream->Read(buffer, DUMMY_SEARCH_SIZE);
395
396 for (size_t i = DUMMY_SEARCH_SIZE - 3; i != 0; --i)
397 {
398 int* sig = (int*) (buffer + i);
399 if (*sig == egg::Magic::DUMMY)
400 {
401 return static_cast<unsigned int> (100 - i);
402 }
403 }
404
405 return 0;
406}

426int EggFormat::PrepareSolidBlock(egg::BlockWrap* block, BlockExtractInfo* info)
427{
428 int res = Result::Success;
429
430 info->coder = NULL;
431 info->dataHeaderSize = 0;
432 info->crc = 0;
433 info->dataRemained = block->GetBlock().GetUncompressedSize();
434 CRC32Init(&info->crc);
435
436 istream_.Seek(block->GetDataOffset(), SEEK_SET);
437
438 GenerateCoder(block->GetBlock().GetCompressMethod(), &info->coder, &info->dataHeaderSize);
439
440 if (info->coder)
441 {
442 unsigned char dataHeader[info->dataHeaderSize];
443 istream_.Read(dataHeader, info->dataHeaderSize);
444 res = info->coder->Initialize(dataHeader, info->dataHeaderSize);
445 }
446 else
447 {
448 res = Result::Unsupported;
449 }
450 info->rawDataRemained = block->GetBlock().GetCompressedSize() - info->dataHeaderSize;
451
452 return res;
453}

455bool EggFormat::IsTargetDir(egg::FileWrap* file)
456{
457 bool res = false;
458 egg::WindowsField *wfield = (egg::WindowsField*) file->GetExtra(egg::Magic::WINDOWS_FILEINFO);
459 if (wfield)
460 {
461 res = wfield->GetAttribute() & 128;
462 }
463 else
464 {
465 egg::PosixField* pfield = (egg::PosixField*) file->GetExtra(egg::Magic::POSIX_FILEINFO);
466 if (pfield)
467 {
468 res = pfield->GetMode() & 0x40000;
469 }
470 }
471
472 return res;
473}

475void EggFormat::GenerateCoder(int method, Coder** coder, size_t* dataHeaderSize)
476{
477 *dataHeaderSize = 0;
478
479 switch (method)
480 {
481 case CompressionMethod::Store:
482 *coder = new StoreCoder();
483 break;
484 case CompressionMethod::Deflate:
485 *coder = new DeflateCoder();
486 break;
487 case CompressionMethod::bzip:
488 *coder = new BZipCoder();
489 break;
490 case CompressionMethod::lzma:
491 *dataHeaderSize = 9;
492 *coder = new LZMACoder();
493 break;
494 case CompressionMethod::azo:
495 *coder = new AZOCoder();
496 break;
497 default:
498 *coder = NULL;
499 break;
500 }

503int EggFormat::ExtractNormalArchive()
504{
505 int res = Result::NotReady;
506
507 if (opened_)
508 {
509 res = OnWorkStart(WorkCode::Extract);
510 if (res == Result::Success)
511 {
512 res = istream_.Seek(0, SEEK_SET);
513
514 if (res == Result::Success)
515 {
516 for (size_t i = 0; i < files_.size(); ++i)
517 {
518 res = OnFileStart(i);
519 if (res == Result::Success)
520 {
521 OutputStream* stream = NULL;
522 res = QueryExtractStream(i, &stream);
523
524 if (res == Result::Success)
525 {
526 res = ExtractIndividualFile(i, stream);
527 }
528 }
529 OnFileFinish(i, res);
530
531 if (res != Result::Skip && res != Result::Success)
532 {
533 break;
534 }
535 }
536
537 istream_.Close();
538 }
539 }
540 OnWorkFinish(WorkCode::Extract, res);
541 }
542
543 return res;

546int EggFormat::ExtractSolidArchive()
547{
548 int res = OnWorkStart(WorkCode::Extract);
549
550 if (res == Result::Success)
551 {
552 res = istream_.Seek(0, SEEK_SET);
553
554 if (res == Result::Success)
555 {
556 BlockExtractInfo info;
557 unsigned char input[egg::EXTRACT_CHUNK] = { 0, };
558 unsigned char output[egg::EXTRACT_CHUNK] = { 0, };
559 size_t blockIndex = 0;
560 egg::BlockWrap* block = solidBlocks_[blockIndex];
561 PrepareSolidBlock(block, &info);
562 if (info.coder)
563 {
564 size_t ucs = 0;
565
566 for (size_t i = 0; i < files_.size(); ++i)
567 {
568 res = OnFileStart(i);
569
570 if (res == Result::Success)
571 {
572 bool skipThisFile = false;
573 egg::FileWrap* file = files_[i];
574 uint64 fileSize = file->GetFileHeader().GetLength();
575 uint64 fileRemained = fileSize;
576 OutputStream* stream = NULL;
577 res = QueryExtractStream(i, &stream);
578 skipThisFile = (res == Result::Skip) || !stream;
579
580 if (res == Result::Cancel)
581 {
582 break;
583 }
584
585 if (res == Result::Success || skipThisFile)
586 {
587 bool isDir = IsTargetDir(file);
588 if (isDir)
589 {
590 OnProgress(i, 0, PROGRESS_MAX);
591 OnProgress(i, PROGRESS_MAX, PROGRESS_MAX);
592 }
593 else
594 {
595 if (skipThisFile || res == Result::Success)
596 {
597 res = OnProgress(i, 0, PROGRESS_MAX);
598
599 if (res == Result::Success)
600 {
601 // Main Loop
602 while (fileRemained)
603 {
604 while (res == Result::Success)
605 {
606 ucs = static_cast<size_t> (egg::EXTRACT_CHUNK > fileRemained ? fileRemained
607 : egg::EXTRACT_CHUNK);
608
609 if (ucs)
610 {
611 res = info.coder->Pop(output, &ucs);
612 ucs = static_cast<size_t> (ucs > info.dataRemained ? info.dataRemained : ucs);
613
614 if (ucs)
615 {
616 if (!skipThisFile)
617 {
618 stream->Write(output, ucs);
619 }
620 info.dataRemained -= ucs;
621 CRC32Update(&info.crc, output, static_cast<unsigned int> (ucs));
622 fileRemained -= ucs;
623 }
624 }
625 else
626 {
627 res = Result::End;
628 }
629 }
630 if (res == Result::End)
631 {
632 res = Result::Success;
633 }
634
635 if (info.rawDataRemained == 0)
636 {
637 CRC32Finish(&info.crc);
638 if (info.crc == block->GetBlock().GetCRC())
639 {
640 res = Result::Success;
641 }
642 else
643 {
644 res = OnError(i, Result::InvalidCRC);
645 }
646
647 if (res == Result::Success)
648 {
649 ++blockIndex;
650 delete info.coder;
651 info.coder = NULL;
652 if (blockIndex < solidBlocks_.size())
653 {
654 block = solidBlocks_[blockIndex];
655 PrepareSolidBlock(block, &info);
656 }
657 else if (fileRemained)
658 {
659 if (fileRemained)
660 {
661 res = OnError(i, Result::InsufficientBlock);
662 }
663 }
664 }
665 }
666
667 if (fileRemained)
668 {
669 size_t
670 readSize =
671 static_cast<size_t> (egg::EXTRACT_CHUNK > info.rawDataRemained ? info.rawDataRemained
672 : egg::EXTRACT_CHUNK);
673 readSize = istream_.Read(input, readSize);
674 if (readSize)
675 {
676 info.coder->Push(input, readSize);
677 info.rawDataRemained -= readSize;
678 }
679 else
680 {
681 res = OnError(i, Result::ReadError);
682 }
683 }
684
685 if (res != Result::Success)
686 {
687 break;
688 }
689 res = OnProgress(i, 10000 - fileRemained * PROGRESS_MAX * 1.0f / fileSize, PROGRESS_MAX);
690 }
691 }
692
693 if (res == Result::Success)
694 {
695 OnProgress(i, PROGRESS_MAX, PROGRESS_MAX);
696 }
697 }
698 else
699 {
700 res = OnError(i, Result::WriteError);
701 }
702
703 if (stream)
704 {
705 stream->Close();
706 delete stream;
707 stream = NULL;
708 }
709 }
710 }
711 else
712 {
713 break;
714 }
715 }
716 OnFileFinish(i, res);
717
718 if (res != Result::Success)
719 {
720 break;
721 }
722 }
723 }
724 else
725 {
726 res = Result::SolidError;
727 }
728
729 if (info.coder)
730 {
731 delete info.coder;
732 info.coder = NULL;
733 }
734 }
735
736 istream_.Close();
737 }
738 OnWorkFinish(WorkCode::Extract, res);
739
740 return res;
741}

743int EggFormat::ExtractIndividualFile(size_t index, OutputStream* stream)
744{
745 int res = Result::Success;
746 egg::FileWrap* file = files_[index];
747 egg::EncryptField* field = (egg::EncryptField*) file->GetExtras().GetExtra(egg::Magic::ENCRYPT);
748
749 bool isDir = IsTargetDir(file);
750
751 if (isDir)
752 {
753 OnProgress(index, 0, PROGRESS_MAX);
754 OnProgress(index, PROGRESS_MAX, PROGRESS_MAX);
755 res = Result::Success;
756 }
757 else
758 {
759 bool isEncrypted = field != NULL;
760
761 if (isEncrypted)
762 {
763 res = PreprocessDecrypt(index, field);
764 }
765
766 if (res == Result::Success)
767 {
768 if (stream)
769 {
770 if (file->GetBlockCount())
771 {
772 for (size_t i = 0; i < file->GetBlockCount(); ++i)
773 {
774 egg::BlockWrap* block = file->GetBlock(i);
775 BlockExtractInfo info;
776 memset(&info, 0, sizeof(info));
777 istream_.Seek(block->GetDataOffset(), SEEK_SET);
778
779 GenerateCoder(block->GetBlock().GetCompressMethod(), &info.coder, &info.dataHeaderSize);
780
781 if (info.coder)
782 {
783 unsigned char dataHeader[info.dataHeaderSize];
784 istream_.Read(dataHeader, info.dataHeaderSize);
785 res = info.coder->Initialize(dataHeader, info.dataHeaderSize);
786
787 res = OnProgress(index, 0, PROGRESS_MAX);
788
789 if (res == Result::Success)
790 {
791 unsigned int crc = 0;
792 CRC32Init(&crc);
793 uint64 size = block->GetBlock().GetCompressedSize() - info.dataHeaderSize;
794 uint64 uncSize = block->GetBlock().GetUncompressedSize();
795 uint64 wholeSize = size;
796 unsigned char input[egg::EXTRACT_CHUNK] = { 0, };
797 unsigned char output[egg::EXTRACT_CHUNK] = { 0, };
798
799 while (size)
800 {
801 size_t readsize = static_cast<size_t> (egg::EXTRACT_CHUNK > size ? size : egg::EXTRACT_CHUNK);
802 readsize = istream_.Read(input, readsize);
803 if (readsize)
804 {
805 if (isEncrypted)
806 {
807 decoder_->DecryptBuffer(input, readsize);
808 }
809 res = info.coder->Push(input, readsize);
810 while (1)
811 {
812 size_t ucs = egg::EXTRACT_CHUNK;
813 res = info.coder->Pop(output, &ucs);
814
815 if (ucs)
816 {
817 ucs = static_cast<size_t> (ucs > uncSize ? uncSize : ucs);
818 stream->Write(output, ucs);
819 CRC32Update(&crc, output, static_cast<unsigned int> (ucs));
820 uncSize -= ucs;
821 }
822
823 if (res == Result::End)
824 {
825 res = Result::Success;
826 break;
827 }
828 }
829 size -= readsize;
830 res = OnProgress(index, 10000 - size * PROGRESS_MAX * 1.0f / wholeSize, PROGRESS_MAX);
831
832 if (res != Result::Success)
833 {
834 break;
835 }
836 }
837 }
838 if (res == Result::Success)
839 {
840 OnProgress(index, PROGRESS_MAX, PROGRESS_MAX);
841 CRC32Finish(&crc);
842 if (crc == block->GetBlock().GetCRC())
843 {
844 res = Result::Success;
845 }
846 else
847 {
848 res = Result::InvalidCRC;
849 }
850
851 if (isEncrypted)
852 {
853 decoder_->PostprocessDecrypt(field->GetAESFooter());
854 if (!decoder_->VerifyNow(NULL))
855 {
856 res = OnError(index, Result::DecryptError);
857 }
858 }
859 }
860 }
861
862 delete info.coder;
863 info.coder = NULL;
864 }
865 else
866 {
867 res = OnError(index, Result::Unsupported);
868 }
869
870 if (res != Result::Success)
871 {
872 res = OnError(index, res);
873 if (res == Result::Success || res == Result::Skip)
874 {
875 continue;
876 }
877 break;
878 }
879 }
880 }
881 else
882 {
883 // 0 byte file
884 OnProgress(index, 0, PROGRESS_MAX);
885 OnProgress(index, PROGRESS_MAX, PROGRESS_MAX);
886 res = Result::Success;
887 }
888 }
889 else
890 {
891 // todo
892 res = OnError(index, Result::WriteError);
893 }
894
895 if (stream)
896 {
897 stream->Close();
898 }
899 }
900 }
901
902 return res;
903}

905int EggFormat::PreprocessDecrypt(size_t index, egg::EncryptField* header)
906{
907 char password[egg::MAX_PASSWORD_LEN];
908 memset(password, 0, sizeof(password));
909
910 decoder_ = NULL;
911 switch (header->GetEncryptMethod())
912 {
913 case EncryptMethod::ZIP_COMPATIBLE:
914 decoder_ = new ZipDecoder();
915 break;
916 case EncryptMethod::AES_128:
917 decoder_ = new AESDecoder(AESDecoder::AES128);
918 break;
919 case EncryptMethod::AES_256:
920 decoder_ = new AESDecoder(AESDecoder::AES256);
921 break;
922 }
923
924 int res = Result::Success;
925 if (decoder_)
926 {
927 res = Result::Success;
928 size_t qc = password_.empty() ? 1 : 0;
929 while (qc < 4)
930 {
931 if (!qc)
932 {
933 strncpy(password, password_.c_str(), egg::MAX_PASSWORD_LEN);
934 }
935 else
936 {
937 res = OnQueryPassword(index, password, egg::MAX_PASSWORD_LEN, qc);
938 }
939
940 if (res == Result::Success)
941 {
942 switch (header->GetEncryptMethod())
943 {
944 case EncryptMethod::ZIP_COMPATIBLE:
945 res = decoder_->PreprocessDecrypt(password, header->GetVerifyData());
946 if (res == Result::Success)
947 {
948 unsigned char vd = header->GetCRC() >> 24;
949 res = decoder_->VerifyNow(&vd) ? Result::Success : OnError(index, Result::DecryptError);
950 }
951 break;
952 case EncryptMethod::AES_128:
953 case EncryptMethod::AES_256:
954 res = decoder_->PreprocessDecrypt(password, header->GetAESHeader());
955 if (res == Result::Success)
956 {
957 res = decoder_->VerifyNow(NULL) ? Result::Success : OnError(index, Result::DecryptError);
