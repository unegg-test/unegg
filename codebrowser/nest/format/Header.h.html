15#include <nest/nest.h>

17#include <nest/stream/InputStream.h>

20namespace nest
21{
22
23#define IF_READ_STREAM_VAR(stream, var) if (sizeof(var) == stream->Read(&var, sizeof(var)))
24#define IF_READ_STREAM_ARY(stream, ary) if (sizeof(ary) == stream->Read(ary, sizeof(ary)))
25#define IF_READ_STREAM_PTR(stream, ptr, size) if (size == stream->Read(ptr, size))
26
27class Header: public Info
28{
29public:
30 Header();
31 virtual ~Header();
32
33 void Clear();
34 int Read(InputStream* stream);
35
36 inline bool IsLoaded()
37 {
38 return loaded_;
39 }
40
41protected:
42 virtual void OnCleanUp() = 0;
43 virtual int OnRead(InputStreamRewinder* rewinder, InputStream* stream) = 0;
44
45private:
46 bool loaded_;
47};
48
49}
