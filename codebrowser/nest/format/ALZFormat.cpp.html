43int ALZFormat::Open(InputStream* stream)
44{
45 if (opened_)
46 {
47 Close();
48 }
49
50 int res = istream_.Initialize(stream->GetVirtualName());
51 istream_.PushStream(stream);
52
53 if (res == Result::Success)
54 {
55 res = istream_.Open();
56
57 if (res == Result::Success)
58 {
59 res = CheckHeaderInternal(stream, &global_);
60 if (res == Result::Success)
61 {
62 res = global_.GetVersion() == 10 ? Result::Success : Result::Invalid;
63 opened_ = res == Result::Success;
64 }
65 }
66 }
67
68 return res;
69}

85int ALZFormat::CheckHeaderInternal(InputStream* stream, alz::Global* global)
86{
87 int res = OnWorkStart(WorkCode::Check);
88
89 if (res == Result::Success)
90 {
91 stream->Seek(0, SEEK_SET);
92 int res = Result::Invalid;
93
94 if (global)
95 {
96 res = global->Read(stream);
97
98 if (res == Result::Success)
99 {
100 res = global->GetVersion() == 10 ? Result::Success : Result::Invalid;
101 }
102 }
103 }
104
105 OnWorkFinish(WorkCode::Check, res);
106
107 return res;
108}

110int ALZFormat::Scan()
111{
112 int res = Result::NotReady;
113
114 if (opened_)
115 {
116 res = OnWorkStart(WorkCode::Scan);
117
118 if (res == Result::Success)
119 {
120 unsigned int signature = 0;
121 istream_.Seek(4, SEEK_END);
122 istream_.Read(&signature, 4);
123 spanned_ = signature == alz::Magic::SPLIT;
124 if (spanned_)
125 {
126 res = PrepareSpannedStreams();
127 }
128
129 if (res == Result::Success)
130 {
131 istream_.Seek(0, SEEK_SET);
132
133 res = ScanFiles();
134 if (res == Result::Success)
135 {
136 res = ScanFooters();
137 }
138 }
139 }
140 OnWorkFinish(WorkCode::Scan, res);
141 }
142
143 return res;
144}

146int ALZFormat::ScanFiles()
147{
148 istream_.Seek(8, SEEK_SET);
149 InputStreamRewinder rewinder(istream_);
150 int res = Result::Invalid;
151
152 for (;;)
153 {
154 alz::File* file = new alz::File();
155 res = file->Read(&istream_);
156
157 if (res == Result::Success)
158 {
159 files_.push_back(file);
160 res = OnFileStart(files_.size() - 1);
161 OnFileFinish(files_.size() - 1, Result::Success);
162
163 if (res != Result::Success)
164 {
165 break;
166 }
167 }
168 else if (res == Result::End)
169 {
170 delete file;
171 res = Result::Success;
172
173 break;
174 }
175 else if (res == Result::NeedMoreStream)
176 {
177 res = Result::Success;
178 }
179 else
180 {
181 delete file;
182 break;
183 }
184 }
185
186 if (res == Result::Success)
187 {
188 rewinder.Clear();
189 }
190
191 return res;
192}

194int ALZFormat::ScanFooters()
195{
196 int res = Result::ReadError;
197 unsigned char endheader[16];
198 istream_.Seek(16, SEEK_END);
199
200 size_t read = istream_.Read(endheader, 16);
201 if (read == 16)
202 {
203 if ((*(unsigned int*) (endheader)) != alz::Magic::END)
204 {
205 res = Result::Invalid;
206 }
207 else
208 {
209 crc_ = *(unsigned int*) (endheader + 8);
210 unsigned int size = *(unsigned int*) (endheader + 4);
211 if (size)
212 {
213 istream_.Seek(size + 16, SEEK_END);
214
215 unsigned int signature = 0;
216 read = istream_.Read(&signature, 4);
217 if (read == 4 && signature == alz::Magic::COMMENT)
218 {
219 size -= 4;
220 unsigned int index = 0;
221 unsigned short len = 0;
222 char comment[65536] = { 0, };
223
224 int64 remained = size;
225 res = Result::Success;
226 while (remained > 6)
227 {
228 read = istream_.Read(&index, 4) + istream_.Read(&len, 2);
229 if (read == 6)
230 {
231 remained -= 6;
232 read = istream_.Read(comment, len);
233 if (read == len)
234 {
235 remained -= len;
236 comment[len] = 0;
237 if (index == 0xFFFFFFFF)
238 {
239 utf8::euckr2utf8(comment, &comment_);
240 }
241 else if (files_.size() > index)
242 {
243 std::string buf;
244 utf8::euckr2utf8(comment, &buf);
245 files_[index]->SetComment(buf.c_str());
246 }
247 else
248 {
249 res = Result::Invalid;
250 break;
251 }
252 }
253 else
254 {
255 break;
256 }
257 }
258 else
259 {
260 break;
261 }
262 }
263 }
264 }
265 else
266 {
267 res = Result::Success;
268 }
269 }
270 }
271
272 return res;
273}

275int ALZFormat::GenerateVolumeName(size_t index, char* buffer, size_t size)
276{
277 std::string name = istream_.GetVirtualName();
278 size_t pos = name.rfind('.');
279 if (pos != std::string::npos)
280 {
281 name = name.substr(0, pos);
282 }
283 char extension[4];
284 if (index == 0)
285 {
286 strcpy(extension, "alz");
287 }
288 else
289 {
290 --index;
291 extension[0] = (index / 100) + 'a';
292 extension[1] = (index % 100) / 10 + '0';
293 extension[2] = (index % 100) % 10 + '0';
294 extension[3] = 0;
295 }
296 snprintf(buffer, size, "%s.%s", name.c_str(), extension);
297
298 return Result::Success;
299}

301int ALZFormat::PrepareSpannedStreams()
302{
303 std::vector<SpannedVolume> volumes;
304 InputStreamRewinder sr(istream_);
305
306 size_t qc = 0;
307 size_t ai = 0;
308
309 int res = Result::Success;
310 while (res == Result::Success)
311 {
312 ++qc;
313 InputStream* stream = NULL;
314 char volumename[FILENAME_LENGTH_MAX] = { 0, };
315 GenerateVolumeName(ai, volumename, FILENAME_LENGTH_MAX);
316 res = OnQueryInputStream(ai, &stream, volumename, qc);
317
318 if (res != Result::Success)
319 {
320 break;
321 }
322
323 if (stream)
324 {
325 stream->Open();
326 alz::Global gh;
327 res = CheckHeaderInternal(stream, &gh);
328
329 if (res == Result::Success)
330 {
331 if (gh.GetDiskID() == ai)
332 {
333 stream->Seek(stream->Size() - 4, SEEK_SET);
334 unsigned int signature = 0;
335 size_t read = stream->Read(&signature, 4);
336 if (4 == read)
337 {
338 SpannedVolume vol;
339 vol.offset = ai ? 8 : 0;
340 vol.stream = stream;
341 vol.size = stream->Size() - vol.offset;
342 switch (signature)
343 {
344 case alz::Magic::FILE_END:
345 res = Result::End;
346 volumes.push_back(vol);
347 break;
348 case alz::Magic::SPLIT:
349 vol.size -= 16;
350 volumes.push_back(vol);
351 qc = 0;
352 ++ai;
353 break;
354 }
355 }
356 }
357 }
358 stream->Close();
359 }
360 }
361
362 if ((res == Result::Success || res == Result::End) && volumes.size())
363 {
364 istream_.Close();
365 istream_.PopAll();
366
367 for (size_t i = 0; i < volumes.size(); ++i)
368 {
369 SpannedVolume& vol = volumes[i];
370 istream_.PushStream(vol.stream, vol.offset, vol.size);
371 }
372
373 istream_.UpdateStreams();
374 res = istream_.Open();
375 }
376
377 return res;
378}

385int ALZFormat::Extract()
386{
387 int res = Result::NotReady;
388
389 if (opened_)
390 {
391 res = OnWorkStart(WorkCode::Extract);
392 if (res == Result::Success)
393 {
394 res = istream_.Seek(0, SEEK_SET);
395
396 if (res == Result::Success)
397 {
398 for (size_t i = 0; i < files_.size(); ++i)
399 {
400 res = OnFileStart(i);
401 if (res == Result::Success)
402 {
403 OutputStream* stream = NULL;
404 res = QueryExtractStream(i, &stream);
405
406 if (res == Result::Success)
407 {
408 res = ExtractIndividualFile(i, stream);
409 }
410 }
411 OnFileFinish(i, res);
412
413 if (res != Result::Skip && res != Result::Success)
414 {
415 break;
416 }
417 }
418
419 istream_.Close();
420 }
421 }
422 OnWorkFinish(WorkCode::Extract, res);
423 }
424
425 return res;
426}

433int ALZFormat::PreprocessDecrypt(size_t index, alz::File* header)
434{
435 char password[alz::MAX_PASSWORD_LEN];
436 memset(password, 0, sizeof(password));
437
438 decoder_ = new ZipDecoder();
439
440 int res = Result::Success;
441 if (decoder_)
442 {
443 res = Result::Success;
444 size_t qc = password_.empty() ? 1 : 0;
445 while (qc < 4)
446 {
447 if (!qc)
448 {
449 strncpy(password, password_.c_str(), alz::MAX_PASSWORD_LEN);
450 }
451 else
452 {
453 res = OnQueryPassword(index, password, alz::MAX_PASSWORD_LEN, qc);
454 }
455
456 if (res == Result::Success)
457 {
458 res = decoder_->PreprocessDecrypt(password, header->GetVerifyData());
459 if (res == Result::Success)
460 {
461 unsigned char vd = header->GetCRC() >> 24;
462 res = decoder_->VerifyNow(&vd) ? Result::Success : OnError(index, Result::DecryptError);
463 }
464 }
465 else
466 {
467 break;
468 }
469 if (res == Result::Success)
470 {
471 password_ = password;
472 break;
473 }
474 ++qc;
475 }
476 }
477 else
478 {
479 res = Result::UnknownDecryptMethod;
480 }
481
482 return res;
483}

490void ALZFormat::GenerateCoder(int method, Coder** coder, size_t* dataHeaderSize)
491{
492 *dataHeaderSize = 0;
493
494 switch (method)
495 {
496 case CompressionMethod::Store:
497 *coder = new StoreCoder();
498 break;
499 case CompressionMethod::Deflate:
500 *coder = new DeflateCoder();
501 break;
502 default:
503 *coder = NULL;
504 break;
505 }
506}

508int ALZFormat::ExtractIndividualFile(size_t index, OutputStream* stream)
509{
510 int res = Result::Success;
511 alz::File* file = files_[index];
512
513 bool isDir = IsTargetDir(file);
514
515 if (isDir)
516 {
517 OnProgress(index, 0, PROGRESS_MAX);
518 OnProgress(index, PROGRESS_MAX, PROGRESS_MAX);
519 res = Result::Success;
520 }
521 else
522 {
523 bool isEncrypted = file->IsEncrypted();
524
525 if (isEncrypted)
526 {
527 res = PreprocessDecrypt(index, file);
528 }
529
530 if (res == Result::Success)
531 {
532 if (file->GetBitFlags())
533 {
534 BlockExtractInfo info;
535 memset(&info, 0, sizeof(info));
536 istream_.Seek(file->GetDataOffset(), SEEK_SET);
537
538 GenerateCoder(file->GetCompressMethod(), &info.coder, &info.dataHeaderSize);
539
540 if (info.coder)
541 {
542 unsigned char dataHeader[info.dataHeaderSize];
543 istream_.Read(dataHeader, info.dataHeaderSize);
544 res = info.coder->Initialize(dataHeader, info.dataHeaderSize);
545
546 res = OnProgress(index, 0, PROGRESS_MAX);
547
548 if (res == Result::Success)
549 {
550 unsigned int crc = 0;
551 CRC32Init(&crc);
552 uint64 size = file->GetPackedSize() - info.dataHeaderSize;
553 uint64 uncSize = file->GetUnpackedSize();
554 uint64 wholeSize = size;
555 unsigned char input[alz::EXTRACT_CHUNK] = { 0, };
556 unsigned char output[alz::EXTRACT_CHUNK] = { 0, };
557
558 while (size)
559 {
560 size_t readsize = static_cast<size_t> (alz::EXTRACT_CHUNK > size ? size : alz::EXTRACT_CHUNK);
561 readsize = istream_.Read(input, readsize);
562 if (readsize)
563 {
564 if (isEncrypted)
565 {
566 decoder_->DecryptBuffer(input, readsize);
567 }
568 res = info.coder->Push(input, readsize);
569 while (1)
570 {
571 size_t ucs = alz::EXTRACT_CHUNK;
572 res = info.coder->Pop(output, &ucs);
573
574 if (ucs)
575 {
576 ucs = static_cast<size_t> (ucs > uncSize ? uncSize : ucs);
577 stream->Write(output, ucs);
578 CRC32Update(&crc, output, static_cast<unsigned int> (ucs));
579 uncSize -= ucs;
580 }
581
582 if (res == Result::End)
583 {
584 res = Result::Success;
585 break;
586 }
587 }
588 size -= readsize;
589 res = OnProgress(index, 10000 - size * PROGRESS_MAX * 1.0f / wholeSize, PROGRESS_MAX);
590
591 if (res != Result::Success)
592 {
593 break;
594 }
595 }
596 }
597 if (res == Result::Success)
598 {
599 OnProgress(index, PROGRESS_MAX, PROGRESS_MAX);
600 CRC32Finish(&crc);
601 if (crc == file->GetCRC())
602 {
603 res = Result::Success;
604 }
605 else
606 {
607 res = Result::InvalidCRC;
608 }
609
610 if (isEncrypted)
611 {
612 decoder_->PostprocessDecrypt(file);
613 if (!decoder_->VerifyNow(NULL))
614 {
615 res = OnError(index, Result::DecryptError);
616 }
617 }
618 }
619 }
620
621 delete info.coder;
622 info.coder = NULL;
623
624 if (res != Result::Success)
625 {
626 res = OnError(index, res);
627 }
628 }
629 else
630 {
631 // todo
632 res = OnError(index, Result::UnknownAlgorithm);
633 }
634 }
635 else
636 {
637 // 0 byte file
638 OnProgress(index, 0, PROGRESS_MAX);
639 OnProgress(index, PROGRESS_MAX, PROGRESS_MAX);
640 res = Result::Success;
641 }
642
643 if (stream)
644 {
645 stream->Close();
646 }
647 }
648 }
649
650 return res;
651}

664{
665 VERIFY_KEY(key::archive::IS_SOLID, 0);
666 VERIFY_KEY(key::archive::IS_SPANNED, spanned_);
667
668 if (!comment_.empty())

670 VERIFY_KEY(key::archive::comment::IS_ENCRYPTED, 0);
671 VERIFY_KEY(key::archive::comment::IS_UTF8, 0);
672 VERIFY_KEY(key::archive::comment::LOCALE, 0);
673 }
674
675 bool handle = false;
676 int res = global_.AsInt32(key, &handle);
677
678 if (handled)
679 {
680 *handled = handle;
681 }
682
683 assert(handled || handle);
684
685 return res;
686}
