33class EggFormat: public Format, Events, Info

40 // Format Implementation
41 virtual int Open(InputStream* stream);
42 virtual void Close();
43 virtual int CheckHeader(InputStream* stream);
44 virtual int Scan();
45 virtual int Extract();
46
47 virtual InputStream* GetInputStream();
48 virtual Info* GetInfo();
49
50protected:
51 // Info Implementation
52 virtual int _As4Bytes(unsigned int key, bool* handled);
53 virtual int64 _As8Bytes(unsigned int key, bool* handled);

58 // Events Implementation
59 virtual int Initialize(Workspace* sender);

62 virtual int OnWorkStart(int workCode);

65 virtual int OnFileStart(size_t index);
66 virtual void OnFileFinish(size_t index, int res);

68 virtual int OnError(size_t index, int errorCode);
69 virtual int OnQueryInputStream(size_t index, InputStream** stream, const char* virtualName, size_t queryCount);

90 int CheckHeaderInternal(InputStream* stream, egg::Global* header);

100 int QueryExtractStream(size_t index, OutputStream** stream);
101 int PrepareSolidBlock(egg::BlockWrap* block, BlockExtractInfo* info);

105 int ExtractNormalArchive();
106 int ExtractSolidArchive();

109 int PreprocessDecrypt(size_t index, egg::EncryptField* header);
110 int PostprocessDecrypt(size_t index, egg::EncryptField* header);
