12#ifndef NEST_EVENTS_H
13#define NEST_EVENTS_H
14
15#include <nest/nest.h>
16
17namespace nest
18{
19
20class InputStream;
21class OutputStream;
22class Workspace;
23
24class Events
25{
26public:
27 virtual ~Events() {};
28 virtual int Initialize(Workspace* sender) = 0;
29 virtual void Finalize(int res) = 0;
30
31 virtual int OnWorkStart(int workCode) = 0;
32 virtual void OnWorkFinish(int workCode, int res) = 0;
33
34 virtual int OnFileStart(size_t index) = 0;
35 virtual void OnFileFinish(size_t index, int res) = 0;
36
37 virtual int OnError(size_t index, int errorCode) = 0;
38 virtual int OnQueryInputStream(size_t index, InputStream** stream, const char* virtualName, size_t queryCount) = 0;
39 virtual int OnQueryOutputStream(size_t index, OutputStream** stream, const char* virtualName) = 0;
40 virtual int OnQueryPassword(size_t index, char* buffer, size_t size, size_t queryCount) = 0;
41 virtual int OnProgress(size_t index, int current, int max) = 0;
42};
43
44}
45
46#endif /* NEST_EVENTS_H */
