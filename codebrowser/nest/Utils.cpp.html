50size_t strlen(const char* s, size_t size)
51{
52 size_t i = 0;
53 size_t j = 0;
54
55 if (s)
56 {
57 while (s[i])
58 {
59 if (size != (size_t) (-1) && i > size)
60 {
61 break;
62 }
63
64 if ((s[i] & 0xc0) != 0x80)
65 {
66 j++;
67 }
68
69 i++;
70 }
71 }
72
73 return j;
74}

81int charlen(char s)
82{
83 int res = 1;
84
85 if (s & 128)
86 {
87 s <<= 1;
88 for (int i = 0; i < 7; ++i)
89 {
90 if (s & 128)
91 {
92 ++res;
93 }
94 else
95 {
96 break;
97 }
98 s <<= 1;
99 }
100 }
101
102 return res;
103}

116const char* prev(const char* s)
117{
118 if (s && *s)
119 {
120 --s;
121 if ((*s & 128) == 0)
122 {
123 return s;
124 }
125
126 while ((*s & 192) == 128)
127 {
128 --s;
129 }
130
131 return s;
132 }
133
134 return NULL;
135}

154bool nth(const char* s, size_t n, char buffer[6], int* len)
155{
156 const char* pos = offset(s, n);
157 if (pos)
158 {
159 memset(buffer, 0, 6);
160 int l = charlen(*pos);
161 if (len)
162 {
163 *len = l;
164 }
165 strncpy(buffer, pos, l);
166
167 return true;
168 }
169
170 return false;
171}

213size_t euckr2utf8(const char* str, std::string* utf8)
214{
215 size_t len = ::strlen(str);
216 if (!len)
217 {
218 utf8->clear();
219
220 return 0;
221 }
222
223 size_t utf8len = 2 * len;
224 char* buffer = (char*) malloc(utf8len);
225 char* utf8start = buffer;
226#ifdef __MINGW32__
227 size_t res = iconv(g_iconv, (const char**) &str, &len, &utf8start, &utf8len);
228#else
229 size_t res = iconv(g_iconv, (char**) &str, &len, &utf8start, &utf8len);
230#endif
231
232 if (res == (size_t) (-1))
233 {
234 utf8->clear();
235 }
236 else
237 {
238 buffer[i2p(buffer, utf8len)] = 0;
239 *utf8 = buffer;
240 }
241 free(buffer);
242
243 return res;
244}

246size_t console_len(const char* str, size_t bytes)
247{
248 char* p = (char*) str;
249 size_t res = 0;
250 while (*p)
251 {
252 if (bytes && bytes < (size_t)(p - str))
253 {
254 break;
255 }
256 ++res;
257 size_t len = charlen(*p);
258 if (len > 1)
259 {
260 ++res;
261 }
262 p += len;
263 }
264
265 return res;
266}

268void limit(char* buffer, size_t bufSize, size_t limitWidth, const char* str)
269{
270 if (limitWidth && bufSize)
271 {
272 --bufSize;
273 char* sp = (char*) str;
274 char* op = buffer;
275 size_t os = 0;
276 size_t cp = 0;
277 while (*sp && cp < limitWidth && os < bufSize)
278 {
279 size_t len = charlen(*sp);
280 size_t width = (len > 1 ? 2 : 1);
281 if (os + len < bufSize && cp + width <= limitWidth)
282 {
283 memcpy(op, sp, len);
284 op += len;
285 os += len;
286 }
287 sp += len;
288 cp += width;
289 }
290 *op = 0;
291 }
292 else if (bufSize)
293 {
294 *buffer = 0;
295 }
296}

311void path_ellipsis(char* buffer, size_t bufSize, size_t limitWidth, const char* str)
312{
313 if (limitWidth && bufSize)
314 {
315 size_t conwidth = console_len(str);
316 bool done = false;
317
318 if (conwidth > limitWidth)
319 {
320 std::vector<_path_token_info> tokens;
321 std::vector<size_t> removed;
322 const char* srcptr = str;
323 size_t old = 0;
324 size_t pos = 0;
325 while (*srcptr)
326 {
327 if (*srcptr == '/')
328 {
329 _path_token_info info;
330 info.pos = old;
331 info.length = pos - old + 1;
332 old = pos + 1;
333 tokens.push_back(info);
334 }
335 size_t len = charlen(*srcptr);
336 srcptr += len;
337 pos += len;
338 }
339
340 if (tokens.size())
341 {
342 std::vector<size_t> calc;
343 calc.resize(tokens.size());
344 for (size_t i = 0; i < tokens.size(); ++i)
345 {
346 calc[i] = i;
347 }
348
349 conwidth += 4;
350 while (calc.size() && conwidth > limitWidth)
351 {
352 size_t p = (size_t) (calc.size() / 2.0f + 0.49f);
353 _path_token_info& token = tokens[calc[p]];
354 conwidth -= console_len(str + token.pos, token.length);
355 removed.push_back(calc[p]);
356 calc.erase(calc.begin() + p);
357 }
358
359 if (conwidth <= limitWidth)
360 {
361 std::sort(removed.begin(), removed.end(), std::less<size_t>());
362 size_t min = removed[0];
363 size_t max = removed.back();
364 size_t minp = tokens[min].pos;
365 size_t maxp = tokens[max].pos + tokens[max].length - 1;
366
367 char* outptr = buffer;
368 size_t srcpos = 0;
369
370 for (size_t i = 0; i < ::strlen(str) && srcpos < bufSize - 1; ++i)
371 {
372 if (i < minp)
373 {
374 *outptr++ = str[i];
375 ++srcpos;
376 }
377 if (i >= minp && i < minp + 3)
378 {
379 *outptr++ = '.';
380 ++srcpos;
381 }
382 if (i >= maxp)
383 {
384 *outptr++ = str[i];
385 ++srcpos;
386 }
387 }
388 *outptr = 0;
389
390 done = true;
391 }
392 }
393 }
394
395 if (!done)
396 {
397 size_t console_width = console_len(str);
398 if (console_width > limitWidth)
399 {
400 const char* src = str;
401 while (console_width + 3 > limitWidth && *src)
402 {
403 size_t len = charlen(*src);
404 console_width -= len > 1 ? 2 : 1;
405 src += len;
406 }
407
408 strncpy(buffer, "...", bufSize < limitWidth ? bufSize : limitWidth);
409 if (bufSize > 3)
410 {
411 strncpy(buffer + 3, src, bufSize - 3);
412 }
413 }
414 else
415 {
416 strncpy(buffer, str, bufSize);
417 }
418 }
419 }
420 else if (bufSize)
421 {
422 *buffer = 0;
423 }
424}

449int mkdir_r(const char *dir, mode_t mode)
450{
451 const char *p = dir + 1;
452
453 while (*p)
454 {
455 if (*p == '/')
456 {
457 *((char*) p) = '\0';
458
459 if (mkdir_(dir, mode) == -1 && errno != EEXIST)
460 return -1;
461
462 *((char*) p) = '/';
463 }
464
465 p++;
466 }
467
468 if (*(p - 1) != '/')
469 if (mkdir_(dir, mode) == -1 && errno != EEXIST)
470 return -1;
471
472 return 0;
473}
