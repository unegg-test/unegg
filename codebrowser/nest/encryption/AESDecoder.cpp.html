25int AESDecoder::OnPreprocess(const char* password, const void* header)
26{
27 int mode = level_ == AES128 ? 1 : 3;
28 unsigned int saltSize = SALT_LENGTH(mode);
29
30 memset(salt_, 0, sizeof(salt_));
31 memcpy(salt_, header, saltSize);
32
33 unsigned char tmp_buf[16] = { 0 };
34 fcrypt_init(mode, (const unsigned char*) password, static_cast<int> (strlen(password)), salt_, tmp_buf, &zcx_);
35
36 verified_ = memcmp(static_cast<const unsigned char*> (header) + saltSize, tmp_buf, PWD_VER_LENGTH) == 0;
37
38 return Result::Success;
39}

41int AESDecoder::OnDecrypt(void* buffer, size_t size)
42{
43 fcrypt_decrypt(static_cast<unsigned char*> (buffer), static_cast<int> (size), &zcx_);
44
45 return Result::Success;
46}

48int AESDecoder::OnPostprocess(const void* footer)
49{
50 unsigned char tmp_buf[16] = { 0 };
51 fcrypt_end(tmp_buf, &zcx_);
52
53 verified_ = memcmp(footer, tmp_buf, 10) == 0;
54
55 return Result::Success;
56}
