25unsigned int ZipDecoder::DecryptByte()
26{
27 unsigned int res = (keys_[2] | 2) & 0xFFFF;
28
29 return ((res * (res ^ 1)) >> 8) & 0xFF;
30}

32void ZipDecoder::InitKeys(const char* password)
33{
34 keys_[0] = 305419896;
35 keys_[1] = 591751049;
36 keys_[2] = 878082192;
37
38 for (size_t i = 0; i < strlen(password); ++i)
39 {
40 UpdateKeys(password[i]);
41 }
42}

44void ZipDecoder::UpdateKeys(unsigned int data)
45{
46 data = data & 0xFF;
47 keys_[0] = g_crc32Table[(keys_[0] ^ data) & 0xFF] ^ (keys_[0] >> 8);
48 keys_[1] = keys_[1] + (keys_[0] & 0xFF);
49 keys_[1] = keys_[1] * 134775813 + 1;
50 keys_[2] = g_crc32Table[(keys_[2] ^ (keys_[1] >> 24)) & 0xFF] ^ ((keys_[2] >> 8) & 0x00FFFFFF);
51}

53int ZipDecoder::OnPreprocess(const char* password, const void* header)
54{
55 InitKeys(password);
56 const unsigned char* p = static_cast<const unsigned char*> (header);
57
58 unsigned int res;
59 for (int i = 0; i < 12; ++i)
60 {
61 res = (p[i] ^ DecryptByte()) & 0xFF;
62 UpdateKeys(res);
63 }
64 verifyData_ = static_cast<unsigned char> (res);
65
66 return Result::Success;
67}

69int ZipDecoder::OnDecrypt(void* buffer, size_t size)
70{
71 unsigned char* p = static_cast<unsigned char*> (buffer);
72
73 for (size_t i = 0; i < size; ++i)
74 {
75 p[i] = p[i] ^ DecryptByte();
76 UpdateKeys(p[i]);
77 }
78
79 return Result::Success;
80}
