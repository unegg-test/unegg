29int DeflateCoder::Initialize(const unsigned char* dataHeader, size_t size)
30{
31 Coder::Initialize(dataHeader, size);
32
33 memset(&stream_, 0, sizeof(stream_));
34 stream_.zalloc = Z_NULL;
35 stream_.zfree = Z_NULL;
36 stream_.opaque = Z_NULL;
37
38 int res = inflateInit2(&stream_, -MAX_WBITS);
39
40 return res == Z_OK ? Result::Success : Result::Fail;
41}

43void DeflateCoder::Finalize()
44{
45 Coder::Finalize();
46
47 inflateEnd(&stream_);
48}

50int DeflateCoder::OnPush(void* inputBuffer, size_t size)
51{
52 stream_.next_in = static_cast<Bytef*> (inputBuffer);
53 stream_.avail_in = static_cast<unsigned int>(size);
54
55 return Result::Success;
56}
57
58int DeflateCoder::OnPop(void* outputBuffer, size_t* size)
59{
60 stream_.avail_out = static_cast<unsigned int>(*size);
61 stream_.next_out = static_cast<Bytef*> (outputBuffer);
62
63 int res = inflate(&stream_, Z_NO_FLUSH);
64
65 bool finished = true;
66 switch (res)
67 {
68 case Z_OK:
69 case Z_STREAM_END:
70 case Z_BUF_ERROR:
71 {
72 finished = stream_.avail_out != 0 || res == Z_STREAM_END;
73
74 *size -= stream_.avail_out;
75 break;
76 }
77 }
78
79 return finished ? Result::End : Result::Success;
80}
