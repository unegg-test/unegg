28int AZOCoder::Initialize(const unsigned char* dataHeader, size_t size)
29{
30 Coder::Initialize(dataHeader, size);
31
32 decoder_ = new AZODecoder();
33 memset(&stream_, 0, sizeof(stream_));
34
35 return Result::Success;
36}
37
38void AZOCoder::Finalize()
39{
40 Coder::Finalize();
41
42 delete decoder_;
43 decoder_ = NULL;
44}

46int AZOCoder::OnPush(void* inputBuffer, size_t size)
47{
48 stream_.next_in = static_cast<char*> (inputBuffer);
49 stream_.avail_in = static_cast<unsigned int> (size);
50
51 return Result::Success;
52}
53
54int AZOCoder::OnPop(void* outputBuffer, size_t* size)
55{
56 stream_.avail_out = static_cast<unsigned int>(*size);
57 stream_.next_out = static_cast<char*> (outputBuffer);
58
59 int res = decoder_->Decompress(stream_);
60
61 bool cont = stream_.avail_out == 0;
62
63 *size -= stream_.avail_out;
64
65 if (res == AZO_STREAM_END)
66 {
67 return Result::End;
68 }
69
70 if (cont)
71 {
72 return Result::Success;
73 }
74
75 return Result::End;
76}
