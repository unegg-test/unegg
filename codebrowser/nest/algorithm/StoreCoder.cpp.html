41int StoreCoder::OnPush(void* inputBuffer, size_t size)
42{
43 buffer_ = inputBuffer;
44 size_ = size;
45
46 return Result::Success;
47}
48
49int StoreCoder::OnPop(void* outputBuffer, size_t* size)
50{
51 *size = size_;
52
53 int res = Result::End;
54
55 if (size_)
56 {
57 memcpy(outputBuffer, buffer_, size_);
58 size_ = 0;
59 res = Result::Success;
60 }
61
62 return res;
63}
