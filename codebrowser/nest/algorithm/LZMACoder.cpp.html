43int LZMACoder::Initialize(const unsigned char* dataHeader, size_t size)
44{
45 Coder::Initialize(dataHeader, size);
46
47 int res = Result::Success;
48 LzmaDec_Construct(&state_);
49 res = LzmaDec_Allocate(&state_, dataHeader + 4, LZMA_PROPS_SIZE, &g_Alloc);
50 if (res == 0)
51 {
52 LzmaDec_Init(&state_);
53 }
54
55 return res;
56}
57
58void LZMACoder::Finalize()
59{
60 Coder::Finalize();
61
62 LzmaDec_Free(&state_, &g_Alloc);
63}

65int LZMACoder::OnPush(void* inputBuffer, size_t size)
66{
67 input_ = static_cast<unsigned char*> (inputBuffer);
68 inputSize_ = size;
69 inputPos_ = 0;
70
71 return Result::Success;
72}

74int LZMACoder::OnPop(void* outputBuffer, size_t* size)
75{
76 unsigned char* obuf = static_cast<unsigned char*> (outputBuffer);
77
78 ELzmaFinishMode finishMode = LZMA_FINISH_ANY;
79 SRes res;
80 size_t inProcessed = inputSize_ - inputPos_;
81 ELzmaStatus status;
82
83 res = LzmaDec_DecodeToBuf(&state_, obuf, size, input_ + inputPos_, &inProcessed, finishMode, &status);
84
85 inputPos_ += inProcessed;
86
87 int r = Result::Fail;
88 switch (status)
89 {
90 case LZMA_STATUS_NOT_FINISHED:
91 r = Result::Success;
92 break;
93 case LZMA_STATUS_NOT_SPECIFIED:
94 r = Result::Fail;
95 break;
96 case LZMA_STATUS_NEEDS_MORE_INPUT:
97 case LZMA_STATUS_FINISHED_WITH_MARK:
98 case LZMA_STATUS_MAYBE_FINISHED_WITHOUT_MARK:
99 r = Result::End;
100 break;
101 }
102 return r;
103}
