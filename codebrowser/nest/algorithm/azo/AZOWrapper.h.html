12#ifndef AZO_AZOWRAPPER_H
13#define AZO_AZOWRAPPER_H
14
15#include "AZO.h"
16
17class AZODecoder
18{
19public:
20 AZODecoder() :
21 handle_(NULL)
22 {
23 AZO_DecompressInit(&handle_);
24 }
25
26 ~AZODecoder()
27 {
28 AZO_DecompressEnd(handle_);
29 }
30
31 int Decompress(const char*& next_in, unsigned int& avail_in,
32 char*& next_out, unsigned int& avail_out)
33 {
34 AZO_Stream stream;
35 stream.next_in = next_in;
36 stream.avail_in = avail_in;
37 stream.next_out = next_out;
38 stream.avail_out = avail_out;
39
40 int ret = Decompress(stream);
41
42 next_in = stream.next_in;
43 avail_in = stream.avail_in;
44 next_out = stream.next_out;
45 avail_out = stream.avail_out;
46
47 return ret;
48 }
49
50 int Decompress(AZO_Stream& stream)
51 {
52 return AZO_Decompress(handle_, &stream);
53 }
54
55private:
56 AZO_HDECOMPRESS handle_;
57};
58
59#endif /*AZO_AZOWRAPPER_H*/
