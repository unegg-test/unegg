137inline u_int Log2(u_int num)
138{
139 u_int log = 0;
140
141 for (; num > 1; ++log)
142 num >>= 1;
143
144 return log;
145}

147inline uint8_t GetMatchLengthCode(u_int value)
148{
149 ASSERT(MATCH_MIN_LENGTH <= value && value <= MATCH_MAX_LENGTH);
150
151 value -= MATCH_MIN_LENGTH;
152
153 if(value < MATCH_LENGTH_SGAP)
154 return static_cast<uint8_t>(value);
155
156 value -= MATCH_LENGTH_SGAP;
157
158 const u_int one(1);
159 const u_int extraBit = Log2(value / MATCH_LENGTH_GAP + 1);
160 const u_int ret = MATCH_LENGTH_SGAP + extraBit*MATCH_LENGTH_GAP
161 + (value - ((one<<extraBit)-1)*MATCH_LENGTH_GAP) / (one<<extraBit);
162
163 return static_cast<uint8_t>(ret);
164}

166inline uint8_t GetMatchDistCode(u_int value)
167{
168 ASSERT(MATCH_MIN_DIST <= value && value <= MATCH_MAX_DIST);
169
170 value -= MATCH_MIN_DIST;
171
172 if(value < MATCH_DIST_SGAP)
173 return static_cast<uint8_t>(value);
174
175 value -= MATCH_DIST_SGAP;
176
177 const u_int one(1);
178 const u_int extraBit = Log2(value / MATCH_DIST_GAP + 1);
179 const u_int ret = MATCH_DIST_SGAP + extraBit*MATCH_DIST_GAP
180 + (value - ((one<<extraBit)-1)*MATCH_DIST_GAP) / (one<<extraBit);
181
182 return static_cast<uint8_t>(ret);
183}
