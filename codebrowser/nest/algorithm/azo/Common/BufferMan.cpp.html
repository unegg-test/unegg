11inline BufferMan::BufferMan() :

18 if(buf_) {

23inline bool BufferMan::Capacity(u_int size)
24{
25 if(size_ < size)
26 {
27 byte* newBuf = static_cast<byte*>(Allocator::Alloc(size));
28 if(newBuf == NULL) {
29 return false;
30 }
31
32 if(GetUseSize()) {
33 ASSERT(buf_);
34 memcpy(newBuf, buf_, static_cast<size_t>(size_));
35 }
36
37 if(buf_) {
38 Allocator::Free(buf_);
39 }
40
41 buf_ = newBuf;
42 size_ = size;
43 }
44
45 return true;
46}

48inline byte* BufferMan::GetBufPtr()
49{
50 ASSERT(wrPtr_ >= rdPtr_);
51
52 if(rdPtr_ != 0) {
53 std::memmove(buf_, buf_ + rdPtr_, static_cast<size_t>(GetUseSize()));
54 wrPtr_ -= rdPtr_;
55 rdPtr_ = 0;
56 }
57
58 return buf_;
59}

61inline u_int BufferMan::Add(const byte* buf, u_int size)
62{
63 u_int s = std::min(GetRemainSize(), size);
64
65 if(size_ - wrPtr_ < size) {
66 GetBufPtr();
67 }
68
69 std::memcpy(buf_+wrPtr_, buf, static_cast<size_t>(s));
70 wrPtr_ += s;
71
72 return s;
73}
74
75inline u_int BufferMan::Ahead(u_int size)
76{
77 ASSERT(size_ - wrPtr_ >= size);
78 wrPtr_ += size;
79
80 return size;
81}
82
83inline u_int BufferMan::Get(byte* buf, u_int size)
84{
85 u_int s = std::min(GetUseSize(), size);
86
87 std::memcpy(buf, buf_+rdPtr_, static_cast<size_t>(s));
88 rdPtr_ += s;
89
90 if(rdPtr_ == wrPtr_) {
91 rdPtr_ = wrPtr_ = 0;
92 }
93
94 return s;
95}
96
97inline u_int BufferMan::Remove(u_int size)
98{
99 u_int s = std::min(GetUseSize(), size);
100
101 rdPtr_ += s;
102
103 if(rdPtr_ == wrPtr_) {
104 rdPtr_ = wrPtr_ = 0;
105 }
106
107 return s;
108}
