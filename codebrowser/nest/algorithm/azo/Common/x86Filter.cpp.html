10const byte CALL (0xE8);
11const byte JMP (0xE9);
12
13inline void NumToByte4(const u_int n, byte* b)
14{
15 b[3] = static_cast<byte>(n >> 24);
16 b[2] = static_cast<byte>(n >> 16);
17 b[1] = static_cast<byte>(n >> 8);
18 b[0] = static_cast<byte>(n);
19}
20
21inline u_int Byte4ToNum(const byte* b)
22{
23 return (static_cast<u_int>(b[3]) << 24) |
24 (static_cast<u_int>(b[2]) << 16) |
25 (static_cast<u_int>(b[1]) << 8) |
26 (static_cast<u_int>(b[0]));
27}
28
29void x86Filter(byte* buf, u_int size, bool encode)
30{
31 if(size < 5)
32 return;
33
34 for(u_int i=0; i<size-4; )
35 {
36 if((buf[i] == CALL) || (buf[i] == JMP))
37 {
38 if(buf[i+4] == 0x00 || buf[i+4] == 0xFF)
39 {
40 u_int addr = Byte4ToNum(buf+i+1);
41
42 if(encode) {
43 addr += i;
44 } else {
45 addr -= i;
46 }
47
48 if((addr >> 24) & 1)
49 addr |= 0xFF<<24;
50 else
51 addr &= (1<<24)-1;
52
53 NumToByte4(addr, buf+i+1);
54 }
55
56 i += 5;
57 } else {
58 ++i;
59 }
60 }
61}
