32template <u_int N>
33inline void EntropyBitProb<N>::IncreProb(u_int pre, BOOL value)
34{
35 ASSERT(pre < ARRAY_N);
36 ASSERT(value < 2);
37
38 const u_int SHIFT_BIT = TOTAL_BIT-6;
39
40 uint32_t& p = prob_[pre];
41 if(value == 0) {
42 p += (TOTAL_COUNT-p)>>SHIFT_BIT;
43 } else {
44 p -= p>>SHIFT_BIT;
45 }
46}

48template <u_int N>
49inline void EntropyBitProb<N>::Update(u_int value)
50{
51 ASSERT(value < N);
52
53 u_int pre(1);
54 for(int i=BIT_N-1; i>=0; --i)
55 {
56 ASSERT(pre < ARRAY_N);
57
58 const u_int v = ((value >> i) & 1);
59 IncreProb(pre, v);
60 pre = (pre << 1) | v;
61 }
62}

64template <u_int N>
65int EntropyBitProb<N>::Compare(EntropyBitProb<N> e1, EntropyBitProb<N> e2, u_int value)
66{
67 uint32_t prob1(1);
68 uint32_t prob2(1);
69
70 u_int pre(1);
71 for(int i=BIT_N-1; i>=0; --i)
72 {
73 const uint32_t p1 = e1.GetProb(pre);
74 const uint32_t p2 = e2.GetProb(pre);
75 const BOOL v = ((value >> i) & 1);
76
77 if((prob1 | prob2) & (((1u<<TOTAL_BIT)-1) << (32-TOTAL_BIT)))
78 {
79 prob1 >>= TOTAL_BIT;
80 prob2 >>= TOTAL_BIT;
81 //ASSERT(prob1);
82 //ASSERT(prob2);
83 }
84
85 prob1 *= (v ? (TOTAL_COUNT-p1) : p1);
86 prob2 *= (v ? (TOTAL_COUNT-p2) : p2);
87
88 pre = (pre << 1) | v;
89 }
90
91 if(prob1 > prob2) {
92 return 1;
93 } else if(prob1 < prob2) {
94 return -1;
95 } else {
96 return 0;
97 }
