30inline const byte* MainCode::GetInBuffer(const byte*& in, u_int& inAvailSize, u_int needSize)
31{
32 if(inBuf_.GetUseSize() >= needSize) {
33 return inBuf_.GetBufPtr();
34 }
35
36 if(inBuf_.GetUseSize() == 0 && inAvailSize >= needSize) {
37 return in;
38 }
39
40 inBuf_.Capacity(needSize);
41
42 u_int size = inBuf_.Add(in, inAvailSize);
43
44 inAvailSize -= size;
45 in += size;
46
47 if(inBuf_.GetUseSize() >= needSize) {
48 return inBuf_.GetBufPtr();
49 } else {
50 return NULL;
51 }
52}

83inline byte* MainCode::GetOutBuffer(byte*& out, u_int& outAvailSize, u_int needSize)
84{
85 if(outBuf_.GetUseSize())
86 {
87 u_int size = outBuf_.Get(out, outAvailSize);
88
89 out += size;
90 outAvailSize -= size;
91 }
92
93 if(outBuf_.GetUseSize()) {
94 return NULL;
95 }
96
97 if(outAvailSize >= outBuf_.GetSize())
98 {
99 ASSERT(outBuf_.GetUseSize() == 0);
100
101 if(outAvailSize >= needSize) {
102 return out;
103 }
104 }
105
106 outBuf_.Capacity(needSize);
107 ASSERT(outBuf_.GetRemainSize() >= needSize);
108
109 return outBuf_.GetBufPtr();
