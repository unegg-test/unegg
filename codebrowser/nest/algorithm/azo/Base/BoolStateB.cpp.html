22template <u_int N>
23inline void BoolState<N>::Update(BOOL b)
24{
25 ASSERT(state_ < ARRAY_N);
26
27 const u_int SHIFT_BIT = TOTAL_BIT-6;
28 uint32_t& p = prob_[state_];
29
30 if(b == 0) {
31 p += (TOTAL_COUNT-p)>>SHIFT_BIT;
32 } else {
33 p -= p>>SHIFT_BIT;
34 }
35
36 state_ = ((state_ << 1) & (ARRAY_N-1)) | b;
37}
