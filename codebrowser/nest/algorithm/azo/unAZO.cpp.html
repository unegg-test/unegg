33//Compress
34int AZO_DecompressInit(AZO_HCOMPRESS* pHandle /*OUT*/)
35{
36 if(pHandle == NULL) {
37 return AZO_PARAM_ERROR;
38 }
39
40 void* buf = Allocator::Alloc(sizeof(Decoder::MainCode));
41 if(buf == NULL) {
42 return AZO_MEM_ERROR;
43 }
44
45 Decoder::MainCode* coder = new (buf) Decoder::MainCode();
46 *pHandle = static_cast<AZO_HCOMPRESS>(coder);
47
48 return AZO_OK;
49}
50
51int AZO_Decompress(AZO_HCOMPRESS handle,
52 AZO_Stream* stream /*INOUT*/)
53{
54 Decoder::MainCode* coder = static_cast<Decoder::MainCode*>(handle);
55
56 if(coder == NULL || stream == NULL) {
57 return AZO_PARAM_ERROR;
58 }
59
60 const byte* in = reinterpret_cast<const byte*>(stream->next_in);
61 u_int iuAvailSize = stream->avail_in;
62 byte* out = reinterpret_cast<byte*>(stream->next_out);
63 u_int outAvailSize = stream->avail_out;
64
65 int ret = coder->Code(in, iuAvailSize, out, outAvailSize);
66
67 stream->next_in = reinterpret_cast<const char*>(in);
68 stream->avail_in = iuAvailSize;
69 stream->next_out = reinterpret_cast<char*>(out);
70 stream->avail_out = outAvailSize;
71
72 return ret;
73}
74
75int AZO_DecompressEnd(AZO_HCOMPRESS handle)
76{
77 Decoder::MainCode* coder = static_cast<Decoder::MainCode*>(handle);
78
79 if(coder == NULL) {
80 return AZO_PARAM_ERROR;
81 }
82
83 coder->~MainCode();
84 Allocator::Free(coder);
85
86 return AZO_OK;
87}
88
89int AZO_BufferDecompress(char* dest,
90 u_int* destLen,
91 const char* source,
92 u_int sourceLen)
93{
94 if(dest == NULL || destLen == NULL || source == NULL) {
95 return AZO_PARAM_ERROR;
96 }
97
98 void* buf = Allocator::Alloc(sizeof(Decoder::MainCode));
99 if(buf == NULL) {
100 return AZO_MEM_ERROR;
101 }
102
103 Decoder::MainCode* coder = new (buf) Decoder::MainCode();
104
105 const byte* in = reinterpret_cast<const byte*>(source);
106 u_int iuAvailSize = sourceLen;
107 byte* out = reinterpret_cast<byte*>(dest);
108 u_int outAvailSize = *destLen;
109
110 int ret = coder->Code(in, iuAvailSize, out, outAvailSize);
111
112 *destLen -= outAvailSize;
113
114 coder->~MainCode();
115 Allocator::Free(coder);
116
117 return ret;
118}
