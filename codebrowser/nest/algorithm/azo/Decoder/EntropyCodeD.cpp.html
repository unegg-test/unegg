27inline void EntropyCode::Rescale()
28{
29 while((low_ & MSB) == (up_ & MSB))
30 {
31 bool b(0);
32 bit_.Code(b);
33 tag_ <<= 1; tag_ |= static_cast<RANGE_TYPE>(b);
34
35 low_ <<= 1;
36 up_ <<= 1; up_ |= 1;
37 }
38
39 while((low_ & sMSB) && (up_ & sMSB) == 0)
40 {
41 bool b(0);
42 bit_.Code(b);
43 tag_ <<= 1; tag_ |= static_cast<RANGE_TYPE>(b);
44 tag_ ^= MSB;
45
46 low_ <<= 1;
47 low_ &= (MSB-1);
48 up_ <<= 1; up_ |= 1;
49 up_ |= MSB;
50 }
51}

53inline u_int EntropyCode::Code(u_int totalBit)
54{
55 ASSERT(init_);
56 ASSERT(low_ < up_);
57
58 RANGE_TYPE t(0);
59 if(low_ == std::numeric_limits<RANGE_TYPE>::min() &&
60 up_ == std::numeric_limits<RANGE_TYPE>::max()) {
61 t = ONE << (std::numeric_limits<RANGE_TYPE>::digits - totalBit);
62 } else {
63 t = (up_ - low_ + 1) >> totalBit;
64 }
65
66 const RANGE_TYPE value = (tag_ - low_) / t;
67
68 up_ = low_ + t * (value+1) - 1;
69 low_ = low_ + t * value;
70
71 Rescale();
72
73 return static_cast<u_int>(value);
74}

76inline BOOL EntropyCode::Code(u_int cumCount, u_int totalBit)
77{
78 ASSERT(init_);
79
80 RANGE_TYPE t(0);
81 if(low_ == std::numeric_limits<RANGE_TYPE>::min() &&
82 up_ == std::numeric_limits<RANGE_TYPE>::max()) {
83 t = ONE << (std::numeric_limits<RANGE_TYPE>::digits - totalBit);
84 } else {
85 t = (up_ - low_ + 1) >> totalBit;
86 }
87
88 const RANGE_TYPE v = (tag_ - low_) / t;
89
90 if(v >= cumCount) {
91 low_ += t * static_cast<RANGE_TYPE>(cumCount);
92 } else {
93 up_ = low_ + t * static_cast<RANGE_TYPE>(cumCount) - 1;
94 }
95 ASSERT(low_ < up_);
96
97 Rescale();
98
99 return v >= cumCount ? 1:0;
100}
