20inline BlockCode::BlockCode(byte* buf, u_int size) :
21 buf_(buf), bufSize_(size),
22 matchCode_(buf, size)

24}

30inline bool BlockCode::CopyBlock(u_int pos, u_int dist, u_int len)
31{
32 if(pos >= dist && pos+len <= bufSize_) {
33 for(u_int i=0; i<len; ++i) {
34 buf_[pos+i] = buf_[pos-dist+i];
35 }
36
37 return true;
38 } else {
39 return false;
40 }
41}

43
44inline u_int BlockCode::GetCode(EntropyCode& entropy, u_int pos)
45{
46 if(matchState_.Code(entropy) == false)
47 {
48 ASSERT(pos > 0);
49
50 buf_[pos] = alphaProb_.Code(entropy, pos, buf_[pos-1]);
51
52 return 1;
53 }
54 else
55 {
56 u_int dist(0);
57 u_int length(0);
58 matchCode_.Code(entropy, pos, dist, length);
59
60 if(CopyBlock(pos, dist, length)) {
61 return length;
62 } else {
63 return 0;
64 }
65 }
66}
67
68inline int BlockCode::Code(EntropyCode& entropy)
69{
70 int ret(AZO_OK);
71
72 entropy.Initialize();
73
74 buf_[0] = alphaProb_.Code(entropy, 0, 0);
75
76 for(u_int i=1; i<bufSize_; )
77 {
78 u_int retLen = GetCode(entropy, i);
79 if(retLen == 0) { //error
80 ret = AZO_DATA_ERROR;
81 break;
82 }
83
84 i += retLen;
85 }
86 entropy.Finalize();
87
88 return ret;
89}
