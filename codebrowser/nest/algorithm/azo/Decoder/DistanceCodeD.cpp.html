21inline u_int DistanceCode::Code(EntropyCode& entropy)
22{
23 u_int dist(0);
24
25 if(history_.Code(entropy, dist) == false)
26 {
27 u_int idxCode = prob_.Code(entropy);
28
29 dist = MATCH_DIST_CODE_TABLE[idxCode];
30 if(MATCH_DIST_EXTRABIT_TABLE[idxCode]) {
31 dist += entropy.Code(MATCH_DIST_EXTRABIT_TABLE[idxCode]);
32 }
33
34 history_.Add(dist);
35 }
36
37 return dist;
38}
