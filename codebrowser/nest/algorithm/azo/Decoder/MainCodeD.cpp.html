25int MainCode::ReadBlock(const byte* in, u_int inSize, byte* out, u_int outSize)
26{
27 if(inSize + COMPRESSION_REDUCE_MIN_SIZE > outSize)
28 {
29 //no compression
30 if(inSize == outSize) {
31 std::memcpy(out, in, static_cast<size_t>(inSize));
32 return AZO_OK;
33 } else {
34 return AZO_DATA_ERROR;
35 }
36 }
37 else
38 {
39 EntropyCode entropy(in, inSize);
40
41 if(coderObj_ == NULL) {
42 u_int s = sizeof(BlockCode);
43 coderObj_ = Allocator::Alloc(s);
44 }
45
46 BlockCode& block = *new (coderObj_)BlockCode(out, outSize);
47
48 int ret = block.Code(entropy);
49 if(entropy.GetSize() != inSize)
50 {
51 ASSERT(ret != AZO_OK);
52 ret = AZO_DATA_ERROR;
53 }
54
55 return ret;
56 }
57}

59int MainCode::Code(const byte*& in, u_int& inAvailSize,
60 byte*& out, u_int& outAvailSize)
61{
62 int ret(AZO_OK);
63
64 for(;;)
65 {
66 if(init_ == false) {
67 const byte* inPtr = GetInBuffer(in, inAvailSize, MAIN_HEAD_SIZE);
68 if(inPtr == NULL)
69 break;
70
71 if(inPtr[0] != '0' + AZO_PRIVATE_VERSION) {
72 ret = AZO_DATA_ERROR_VERSON;
73 break;
74 }
75 useFilter_ = inPtr[1] & 1 ? true : false;
76
77 RemoveInBuffer(in, inAvailSize, MAIN_HEAD_SIZE);
78
79 init_ = true;
80
81 continue;
82 }
83
84 if(setSizeInfo_ == false)
85 {
86 const byte* inPtr = GetInBuffer(in, inAvailSize, BLOCK_HEAD_SIZE);
87 if(inPtr == NULL)
88 break;
89
90 ReadNumber(blockSize_, inPtr, BLOCK_SIZE_SIZE);
91 ReadNumber(compressSize_, inPtr+BLOCK_SIZE_SIZE, BLOCK_SIZE_SIZE);
92
93 u_int checkSize(0);
94 ReadNumber(checkSize, inPtr+BLOCK_SIZE_SIZE*2, BLOCK_SIZE_SIZE);
95
96 RemoveInBuffer(in, inAvailSize, BLOCK_HEAD_SIZE);
97
98 if(blockSize_ < compressSize_ || (blockSize_^compressSize_) != checkSize) {
99 ret = AZO_DATA_ERROR_BLOCKSIZE;
100 break;
101 }
102
103 setSizeInfo_ = true;
104
105 continue;
106 }
107
108 if(finish_)
109 {
110 if(OutBufferUseSize() == 0) {
111 ret = AZO_STREAM_END;
112 } else {
113 GetOutBuffer(out, outAvailSize, 0);
114 }
115 break;
116 }
117
118 if(blockSize_ && compressSize_)
119 {
120 const byte* inPtr = GetInBuffer(in, inAvailSize, compressSize_);
121 byte* outPtr = GetOutBuffer(out, outAvailSize, blockSize_);
122 if(inPtr == NULL || outPtr == NULL)
123 break;
124
125 ret = ReadBlock(inPtr, compressSize_, outPtr, blockSize_);
126
127 if(ret >= AZO_OK) {
128 if(useFilter_) {
129 x86Filter(outPtr, blockSize_, false);
130 }
131
132 setSizeInfo_ = false;
133 RemoveInBuffer(in, inAvailSize, compressSize_);
134 RemoveOutBuffer(out, outAvailSize, blockSize_);
135 } else {
136 break;
137 }
138 }
139 else
140 {
141 finish_ = true;
142 }
143 }
144
145 return ret;
146}
