20inline BitCode::BitCode(const TYPE* buf, u_int size) : inBuf_(buf), bufSize_(size*8)
21{
22 remainBit_ = TYPE_BIT_SIZE;
23 readSize_ = 0;
24}

27template <typename T>
28inline bool BitCode::Code(T& value, u_int bitSize)
29{
30// ASSERT(bufSize_ >= readSize_ + bitSize);
31 if(bufSize_ < readSize_ + bitSize) {
32 return false;
33 }
34
35 readSize_ += bitSize;
36 value = 0;
37
38 if(remainBit_ <= bitSize)
39 {
40 bitSize -= remainBit_;
41 value = static_cast<T>((*inBuf_++ & ((1<<remainBit_)-1)) << bitSize);
42 remainBit_ = TYPE_BIT_SIZE;
43
44 while(bitSize >= TYPE_BIT_SIZE)
45 {
46 bitSize -= TYPE_BIT_SIZE;
47 value |= (*inBuf_++ & TYPE_BIT_MASK) << bitSize;
48 }
49 }
50
51 if(bitSize)
52 {
53 remainBit_ -= bitSize;
54 value |= (*inBuf_ >> remainBit_) & ((1<<bitSize)-1);
55 }
56
57 return true;
58}

62template <typename T>
63inline bool BitCode::Code(T* values, u_int bitSize)
64{
65 for(u_int i=0; i<bitSize/8; ++i) {
66 if(Code(values[i], sizeof(T)*8) == false) {
67 return false;
68 }
69 }
70
71 int r = bitSize - bitSize/8*8;
72 if(r)
73 {
74 if(Put(values[bitSize/8+1], r) == false) {
75 return false;
76 }
77 }
78
79 return true;
80}

83inline bool BitCode::Code(bool& v)
84{
85 //return Code(v, 1);
86
87 ASSERT(readSize_ <= bufSize_);
88 if(bufSize_ == readSize_) {
89 return false;
90 }
91
92 ++readSize_;
93 --remainBit_;
94
95 v = *inBuf_ & (1u << remainBit_) ? true:false;
96
97 if(remainBit_ == 0) {
98 ++inBuf_;
99 remainBit_ = TYPE_BIT_SIZE;
100 }
101
102 return true;
103}
