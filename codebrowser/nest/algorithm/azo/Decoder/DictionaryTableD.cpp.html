20inline bool DictionaryTable::Code(EntropyCode& entropy, u_int& pos, u_int& length)
21{
22 if(findState_.Code(entropy))
23 {
24 u_int n = prob_.Code(entropy);
25 if(n >= N) {
26 return false;
27 }
28
29 Get(n, pos, length);
30
31 Update(pos, length, n);
32
33 return true;
34 }
35 else
36 {
37 return false;
38 }
39}
40
41inline void DictionaryTable::Add(u_int pos, u_int length)
42{
43 Update(pos, length);
44}
45
46inline void DictionaryTable::Get(u_int idx, u_int& pos, u_int& len)
47{
48 ASSERT(idx < N);
49
50 len = data_[idx].len;
51 pos = data_[idx].pos;
52}
