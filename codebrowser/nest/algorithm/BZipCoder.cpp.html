28int BZipCoder::Initialize(const unsigned char* dataHeader, size_t size)
29{
30 Coder::Initialize(dataHeader, size);
31
32 memset(&stream_, 0, sizeof(stream_));
33
34 int res = BZ2_bzDecompressInit(&stream_, 0, 0);
35
36 return res == BZ_OK ? Result::Success : Result::Fail;
37}
38
39void BZipCoder::Finalize()
40{
41 Coder::Finalize();
42
43 BZ2_bzDecompressEnd(&stream_);
44}

46int BZipCoder::OnPush(void* inputBuffer, size_t size)
47{
48 stream_.next_in = static_cast<char*> (inputBuffer);
49 stream_.avail_in = static_cast<unsigned int>(size);
50
51 return Result::Success;
52}
53
54int BZipCoder::OnPop(void* outputBuffer, size_t* size)
55{
56 stream_.avail_out = static_cast<unsigned int>(*size);
57 stream_.next_out = static_cast<char*> (outputBuffer);
58
59 int res = BZ2_bzDecompress(&stream_);
60 int result = Result::Invalid;
61
62 switch (res)
63 {
64 case BZ_OK:
65 case BZ_STREAM_END:
66 *size -= stream_.avail_out;
67
68 if (res == BZ_STREAM_END)
69 {
70 result = Result::End;
71 }
72 else
73 {
74 result = stream_.avail_out == 0 ? Result::Success : Result::End;
75 }
76 break;
77 default:
78 *size = 0;
79 }
80
81 return result;
82}
