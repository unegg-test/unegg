50inline static void CRC32Init(unsigned int *pCrc32)
51{
52 *pCrc32 = 0xFFFFFFFF;
53}
54
55inline static void CRC32Update(unsigned int *pCrc32, const unsigned char* lpData, const unsigned int uSize)
56{
57 unsigned int i = 0;
58
59 for (i = 0; i < uSize; i++)
60 {
61 *pCrc32 = ((*pCrc32) >> 8) ^ g_crc32Table[(lpData[i]) ^ ((*pCrc32) & 0x000000FF)];
62 }
63}
64
65inline static void CRC32Finish(unsigned int *pCrc32)
66{
67 *pCrc32 = ~(*pCrc32);
68}
