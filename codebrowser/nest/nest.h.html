25#define DEFINE_CATEGORY(category) \
26 namespace key \
27 {\
28 namespace category \
29 {\
30 static unsigned int OBJECT = _CHILD++ << 4; \
31 static unsigned int _NEXT = 0; \
32 }\
33 }
34
35#define DEFINE_CATEGORY2(category, category2) \
36 namespace key \
37 {\
38 namespace category \
39 {\
40 namespace category2 \
41 {\
42 static unsigned int OBJECT = _CHILD++ << 4; \
43 static unsigned int _NEXT = 0; \
44 }\
45 }\
46 }

48#define DEFINE_KEY(category, name) \
49 namespace key\
50 {\
51 namespace category \
52 {\
53 static const unsigned int name = OBJECT | ++_NEXT; \
54 }\
55 }
56
57#define DEFINE_KEY2(category, category2, name) \
58 namespace key \
59 {\
60 namespace category \
61 {\
62 namespace category2 \
63 {\
64 static const unsigned int name = OBJECT | ++_NEXT; \
65 }\
66 }\
67 }
