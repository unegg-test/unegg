53void EventHandler::Finalize(int res)
54{
55 printf("\n");
56
57 if (info_->AsBool(key::archive::IS_SOLID) && res == Result::InvalidCRC)
58 {
59 printf("\nSolid CRC is broken.\n");
60 }
61
62 sender_ = NULL;
63 info_ = NULL;
64 filename_.clear();
65 archiveName_.clear();
66}

68int EventHandler::OnWorkStart(int workCode)
69{
70 if (vLevel_ > 1)
71 {
72 switch (workCode)
73 {
74 case WorkCode::Open:
75 PrintIndent(sender_->GetWorkStack().size());
76 printf("Opening archive \"%s\".\n", sender_->GetFilename());
77 break;
78 case WorkCode::Scan:
79 PrintIndent(sender_->GetWorkStack().size());
80 printf("Scanning files in \"%s\".\n", sender_->GetFilename());
81 break;
82 case WorkCode::Extract:
83 PrintIndent(sender_->GetWorkStack().size());
84 printf("Extracting files in \"%s\".\n", sender_->GetFilename());
85 break;
86 }
87 }
88
89 return Result::Success;
90}

114int EventHandler::OnFileStart(size_t index)
115{
116 Info* info = info_->AsChildInfo(key::file::OBJECT, index);
117
118 filename_ = info->AsString(key::file::name::NAME);
119
120 if (sender_->GetLastWork() != WorkCode::Extract)
121 {
122 if (vLevel_ > 1)
123 {
124 printf("Found - %s\n", utf8::path_ellipsis(console_->ws_col - 9, filename_.c_str()));
125 }
126 }
127 else
128 {
129 if (redirect_)
130 {
131 printf("%s ", filename_.c_str());
132 }
133 else
134 {
135 printf(" 0 %% - %s", utf8::path_ellipsis(console_->ws_col - 10, filename_.c_str()));
136 }
137 }
138
139 return Result::Success;
140}

168 printf(" - error.\n");
169 }

180int EventHandler::OnError(size_t index, int errorCode)
181{
182 int res = errorCode;
183
184 std::string msg;

231int EventHandler::OnQueryOutputStream(size_t index, OutputStream** stream, const char* virtualName)
232{
233 int res = Result::Success;
234 bool exists = 0 == access(virtualName, F_OK);
235
236 if (exists)
237 {
238 switch (overwrite_)
239 {
240 case 0:
241 {
242 retry: printf("\n");
243 printf("\"%s\" File already exists. Overwrite?\n(Yes[Y]/No[N]/Yes All[A]/No All[O]/Cancel[C])", filename_.c_str());
244
245 char answer[1024];
246 if (fgets(answer, 1024, stdin))
247 {
248 switch (answer[0])
249 {
250 case 'Y':
251 case 'y':
252 break;
253 case 'N':
254 case 'n':
255 res = Result::Skip;
256 break;
257 case 'A':
258 case 'a':
259 overwrite_ = 1;
260 res = Result::Success;
261 break;
262 case 'O':
263 case 'o':
264 overwrite_ = 2;
265 res = Result::Skip;
266 break;
267 case 'C':
268 case 'c':
269 res = Result::Cancel;
270 break;
271 default:
272 goto retry;
273 }
274 }
275 else
276 {
277 goto retry;
278 }
279 printf("\n");
280 }
281 break;
282 case 1:
283 res = Result::Success;
284 break;
285 case 2:
286 res = Result::Skip;
287 break;
288 }
289 }
290
291 return res;
292}

294int EventHandler::OnQueryPassword(size_t index, char* buffer, size_t size, size_t queryCount)
295{
296 printf("\nEnter password: ");
297 if (fgets(buffer, static_cast<unsigned int> (size), stdin))
298 {
299 size_t len = strlen(buffer);
300 while (len > 0 && buffer[len - 1] == '\n')
301 {
302 buffer[len - 1] = 0;
303 --len;
304 }
305
306 return Result::Success;
307 }
308 return Result::Fail;
309}

316int EventHandler::OnProgress(size_t index, int current, int max)
317{
318 if (!redirect_)
319 {
320 printf("\r");
321 printf("%3d %% - %s", static_cast<int> (current * 100.f / max), utf8::path_ellipsis(console_->ws_col - 10, filename_.c_str()));
322 fflush(stdout);
323 }
324
325 return Result::Success;
326}
