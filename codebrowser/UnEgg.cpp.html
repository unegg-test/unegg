84void PrintError(int error)
85{
86 if (g_verbose > 0)
87 {
88 switch (error)
89 {
90 case Result::Cancel:
91 printf("Canceled by user.\n");
92 break;
93 case Result::InvalidCRC:
94 printf("CRC error.\n");
95 break;
96 default:
97 printf("Error (%d)\n", error);
98 }
99 }
100}

102int PrintListOfArchive(const char* filename)
103{
104 Workspace archive(&g_event);
105
106 int res = archive.Open(filename);
107 Info* info = archive.GetInfo();
108
109 if (res == Result::Success)
110 {
111 size_t headerCount = sizeof(HEADERS) / sizeof(DisplayInfo);
112 size_t restOfLine = g_console.ws_col - 2;
113
114 char splitter[g_console.ws_col];
115 memset(splitter, 0, sizeof(splitter));
116 memset(splitter, '-', g_console.ws_col - 1);
117
118 if (g_verbose > 0)
119 {
120 printf("Global informations\n");
121 printf("[\n");
122 printf("\tFormat : %s\n", info->AsString(key::archive::format::NAME));
123 printf("\tVersion : %d.%d\n", info->AsInt32(key::archive::version::MAJOR), info->AsInt32(key::archive::version::MINOR));
124 bool isSolid = info->AsBool(key::archive::IS_SOLID);
125 printf("\tSolid : %s\n", isSolid ? "yes" : "no");
126 if (isSolid)
127 {
128 printf("\t[\n");
129 printf("\t\tSolid Block Count : %zu\n", info->AsSize(key::archive::BLOCK_COUNT));
130 printf("\t]\n");
131 }
132 printf("\tSpanned : %s\n", info->AsBool(key::archive::IS_SPANNED) ? "yes" : "no");
133 printf("\tFile Count : %zu\n", info->AsSize(key::archive::FILE_COUNT));
134
135 bool handle = false;
136 const char* comment = info->AsString(key::archive::comment::COMMENT, &handle);
137 if (comment)
138 {
139 printf("\tGlobal Comment : %s\n", comment);
140 }
141 else
142 {
143 printf("\tGlobal Comment : none\n");
144 }
145 printf("]\n\n");
146
147 if (g_verbose < 2)
148 {
149 printf("%s\n", splitter);
150 bool isFirst = true;
151 for (size_t i = 1; i < headerCount; ++i)
152 {
153 if (g_disp & ((int) pow(2, i)))
154 {
155 PRINTF2("%c%%%ds", isFirst ? ' ' : g_delimiter, HEADERS[i].length, HEADERS[i].title);
156 restOfLine -= HEADERS[i].length + (isFirst ? 0 : 1);
157 isFirst = false;
158 }
159 }
160 if (g_disp & 1)
161 {
162 PRINTF("%c%%s", isFirst ? ' ' : g_delimiter, HEADERS[0].title);
163 restOfLine -= isFirst ? 0 : 1;
164 }
165 printf("\n%s\n", splitter);
166 }
167 }
168
169 for (size_t i = 0; i < info->AsSize(key::archive::FILE_COUNT); ++i)
170 {
171 Info* finfo = info->AsChildInfo(key::file::OBJECT, i);
172
173 if (g_verbose < 2)
174 {
175 bool isFirst = true;
176 for (size_t j = 1; j < headerCount; ++j)
177 {
178 if (g_disp & (int) pow(2, j))
179 {
180 char fmt[20] = { 0, };
181 snprintf(fmt, sizeof(fmt), "%c%%%d%s", isFirst ? ' ' : g_delimiter, HEADERS[j].length, HEADERS[j].type);
182
183 switch (j)
184 {
185 case 1:
186 printf(fmt, finfo->AsUInt64(key::file::size::PACKED));
187 break;
188 case 2:
189 printf(fmt, finfo->AsUInt64(key::file::size::UNPACKED));
190 break;
191 case 3:
192 printf(fmt, finfo->AsUInt32(key::file::attribute::ATTRIBUTES));
193 break;
194 case 4:
195 printf(fmt, finfo->AsSize(key::file::BLOCK_COUNT));
196 break;
197 case 5:
198 printf(fmt, finfo->AsBool(key::file::IS_ENCRYPTED) ? "yes" : "no");
199 break;
200 case 6:
201 printf(fmt, finfo->AsString(key::file::method::NAME));
202 break;
203 case 7:
204 printf(fmt, finfo->AsUInt32(key::file::CRC));
205 break;
206 case 8:
207 {
208 bool handle = false;
209 const char* comment = finfo->AsString(key::file::comment::COMMENT, &handle);
210 printf(fmt, handle && comment && *comment ? comment : "none");
211 }
212 break;
213 }
214 isFirst = false;
215 }
216 }
217 if (g_disp & 1)
218 {
219 PRINTF("%c%%s", isFirst ? ' ' : g_delimiter, utf8::path_ellipsis(restOfLine, finfo->AsString(key::file::name::NAME)));
220 }
221 }
222 else
223 {
224 printf("%s\n", utf8::path_ellipsis(restOfLine, finfo->AsString(key::file::name::NAME)));
225 printf("[\n");
226 for (size_t j = 1; j < headerCount; ++j)
227 {
228 char fmt[20];
229 snprintf(fmt, sizeof(fmt), "\t%%11s : %%%s\n", HEADERS[j].type);
230
231 if (g_disp & (int) pow(2, j))
232 {
233 switch (j)
234 {
235 case 1:
236 printf(fmt, HEADERS[j].title, finfo->AsUInt64(key::file::size::PACKED));
237 break;
238 case 2:
239 printf(fmt, HEADERS[j].title, finfo->AsUInt64(key::file::size::UNPACKED));
240 break;
241 case 3:
242 printf(fmt, HEADERS[j].title, finfo->AsUInt32(key::file::attribute::ATTRIBUTES));
243 break;
244 case 4:
245 printf(fmt, HEADERS[j].title, finfo->AsSize(key::file::BLOCK_COUNT));
246 break;
247 case 5:
248 printf(fmt, HEADERS[j].title, finfo->AsBool(key::file::IS_ENCRYPTED) ? "yes" : "no");
249 break;
250 case 6:
251 printf(fmt, HEADERS[j].title, finfo->AsString(key::file::method::NAME));
252 break;
253 case 7:
254 printf(fmt, HEADERS[j].title, finfo->AsUInt32(key::file::CRC));
255 break;
256 case 8:
257 {
258 bool handle = false;
259 const char* comment = finfo->AsString(key::file::comment::COMMENT, &handle);
260 printf(fmt, HEADERS[j].title, handle && comment && *comment ? comment : "none");
261 }
262 break;
263 }
264 }
265 }
266 printf("]\n");
267 }
268 printf("\n");
269 }
270 printf("%s\n", splitter);
271 }
272 else
273 {
274 PrintError(res);
275 }
276
277 return res;
278}

280int ExtractArchive(CommandLine& cmd)
281{
282 Workspace archive;
283 archive.SetEvent(&g_event);
284
285 int res = archive.Open(cmd.GetArgument(0));
286
287 if (res == Result::Success)
288 {
289 char cwd[1024] = { 0, };
290 if (cmd.GetArgumentCount() == 1)
291 {
292 if (!getcwd(cwd, 1024))
293 {
294 }
295 }
296 else
297 {
298 strcpy(cwd, cmd.GetArgument(1));
299 }
300 res = archive.Extract(cwd);
301
302 if (res == Result::Success)
303 {
304 if (g_verbose > 0)
305 {
306 printf("Total: %u File(s)\n\n", static_cast<unsigned int> (archive.GetInfo()->AsSize(key::archive::FILE_COUNT)));
307 }
308 }
309 else
310 {
311 PrintError(res);
312 }
313 }
314 else
315 {
316 PrintError(res);
317 }
318
319 return res;
320}

322int ProcessCommandline(CommandLine& cmd)
323{
324 int res = -1;
325
326 if (cmd.CommandExists("x"))
327 {
328 return ExtractArchive(cmd);
329 }
330 else
331 {
332 return PrintListOfArchive(cmd.GetArgument(0));
333 }
334
335 return res;
336}

338int main(int argc, char* argv[])
339{
340 using namespace nest;
341 nest::utf8::initialize();
342 int res = -1;
343
344 setlocale(LC_ALL, "");
345
346#ifndef __MINGW32__
347 ioctl(0, TIOCGWINSZ, &g_console);
348#else
349 g_console.ws_col = 80;
350#endif
351 if (g_console.ws_col == 0)
352 {
353 g_console.ws_col = 132;
354 }
355 nest::Workspace::AddFactory(new nest::EggFormatFactory());
356 nest::Workspace::AddFactory(new nest::ALZFormatFactory());
357
358 CommandLine cmd(argc, argv);
359 if (cmd.GetArgumentCount() < 1 || cmd.GetCommandCount() > 1)
360 {
361 PrintProgramDetails();
362
363 PrintUsage();
364 }
365 else
366 {
367 bool redirect = cmd.CommandExists("r");
368
369 g_event.SetDisplayLevel(g_verbose);
370 g_event.SetRedirect(redirect);
371
372 std::set<std::string>::iterator it;
373 for (it = cmd.GetCommands().begin(); it != cmd.GetCommands().end(); ++it)
374 {
375 std::string str = *it;
376 switch (str[0])
377 {
378 case 'd':
379 {
380 g_disp = 0;
381 for (size_t i = 1; i < str.length(); ++i)
382 {
383 switch (str[i])
384 {
385 case 'n':
386 g_disp |= 1;
387 break;
388 case 'p':
389 g_disp |= 2;
390 break;
391 case 'u':
392 g_disp |= 4;
393 break;
394 case 'a':
395 g_disp |= 8;
396 break;
397 case 'b':
398 g_disp |= 16;
399 break;
400 case 'e':
401 g_disp |= 32;
402 break;
403 case 'm':
404 g_disp |= 64;
405 break;
406 case 'r':
407 g_disp |= 128;
408 break;
409 case 'c':
410 g_disp |= 256;
411 }
412 }
413 }
414 break;
415 case 'p':
416 if (str.length() > 1)
417 {
418 g_delimiter = str[1];
419 }
420 break;
421 case 'v':
422 if (str.length() > 1)
423 {
424 g_verbose = atoi(str.c_str() + 1);
425 }
426 }
427 }
428
429 PrintProgramDetails();
430
431 res = ProcessCommandline(cmd);
432 }
433
434 nest::utf8::finalize();
435
436 return res;
437}
