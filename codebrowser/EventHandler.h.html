15#include <string>

19#ifndef __MINGW32__
20#include <sys/ioctl.h>

22struct winsize
23{
24 unsigned short ws_row; /* rows, in characters */
25 unsigned short ws_col; /* columns, in characters */
26 unsigned short ws_xpixel; /* horizontal size, pixels */
27 unsigned short ws_ypixel; /* vertical size, pixels */
28};
29#endif
30
31class EventHandler: public nest::Events
32{
33public:
34 EventHandler(winsize* console);
35 virtual ~EventHandler();
36
37 void SetDisplayLevel(int vLevel);
38 void SetRedirect(bool redirect);
39
40 virtual int Initialize(nest::Workspace* sender);
41 virtual void Finalize(int res);
42
43 virtual int OnWorkStart(int workCode);
44 virtual void OnWorkFinish(int workCode, int res);
45
46 virtual int OnFileStart(size_t index);
47 virtual void OnFileFinish(size_t index, int res);
48
49 virtual int OnError(size_t index, int errorCode);
50 virtual int OnQueryInputStream(size_t index, nest::InputStream** stream,
51 const char* virtualName, size_t queryCount);
52 virtual int OnQueryOutputStream(size_t index, nest::OutputStream** stream,
53 const char* virtualName);
54 virtual int OnQueryPassword(size_t index, char* buffer, size_t size,
55 size_t queryCount);
56 virtual int OnProgress(size_t index, int current, int max);
57
58private:
59 void PrintIndent(size_t size);
60
61 int vLevel_;
62 bool redirect_;
63 nest::Workspace* sender_;
64 int overwrite_;
65 std::string filename_;
66 std::string archiveName_;
67 nest::Info* info_;
68 winsize* console_;
69};

71#endif /* EVENTHANDLER_H_ */
